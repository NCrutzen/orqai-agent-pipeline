-- Phase 55: migrate debtor.agent_runs → public.agent_runs with swarm_type discriminator.
-- Sales-swarm lands in ~5 days; abstracting on day 1 avoids a painful later migration.
-- See .planning/phases/55-debtor-email-pipeline-hardening/55-CONTEXT.md §agent_runs schema.

-- 1. If debtor.agent_runs has data, copy it first (rare: swarm not launched per STATE.md).
do $$
declare
  src_exists bool;
  row_count int := 0;
begin
  select exists(
    select 1 from information_schema.tables
    where table_schema = 'debtor' and table_name = 'agent_runs'
  ) into src_exists;
  if src_exists then
    execute 'select count(*) from debtor.agent_runs' into row_count;
    raise notice 'debtor.agent_runs has % row(s); will copy to public.agent_runs with swarm_type=''debtor_email''', row_count;
  end if;
end$$;

-- 2. Create public.agent_runs (generic discriminator-based shape).
create table if not exists public.agent_runs (
  id                 uuid primary key default gen_random_uuid(),
  swarm_type         text not null check (swarm_type in ('debtor_email', 'sales_email')),
  email_id           uuid,
  automation_run_id  uuid,
  inngest_run_id     text,

  -- Intent-agent output
  intent             text,
  sub_type           text,
  document_reference text,
  language           text,
  confidence         text check (confidence is null or confidence in ('low','medium','high')),
  urgency            text check (urgency is null or urgency in ('low','normal','high')),
  intent_version     text,
  reasoning          text,

  -- Body-agent output
  body_version       text,
  detected_tone      text,

  -- Tool outputs
  tool_outputs       jsonb not null default '{}'::jsonb,
  draft_url          text,

  -- Kanban state machine
  status             text not null default 'classifying',

  -- HITL verdict
  human_verdict      text check (human_verdict is null or human_verdict in (
    'approved','edited_minor','edited_major',
    'rejected_wrong_intent','rejected_wrong_reference','rejected_wrong_attachment',
    'rejected_wrong_language','rejected_wrong_tone','rejected_other'
  )),
  human_notes        text,
  verdict_reason     text,
  verdict_set_at     timestamptz,
  verdict_set_by     text,

  -- Swarm-specific blob (debtor: {graph_message_id, icontroller_draft_url, entity})
  context            jsonb not null default '{}'::jsonb,

  created_at         timestamptz not null default now(),
  updated_at         timestamptz not null default now(),
  completed_at       timestamptz,

  constraint agent_runs_email_id_fk
    foreign key (email_id) references email_pipeline.emails(id) on delete cascade,
  constraint agent_runs_automation_run_id_fk
    foreign key (automation_run_id) references public.automation_runs(id) on delete set null
);

-- 3. Copy legacy rows (if any) with swarm_type stamped.
do $$
begin
  if exists(select 1 from information_schema.tables where table_schema='debtor' and table_name='agent_runs') then
    insert into public.agent_runs (
      id, swarm_type, email_id, inngest_run_id,
      intent, sub_type, document_reference, language, confidence, urgency,
      intent_version, reasoning, body_version, detected_tone,
      tool_outputs, draft_url, status,
      human_verdict, human_notes, verdict_set_at, verdict_set_by,
      context, created_at, updated_at, completed_at
    )
    select
      id, 'debtor_email', email_id, inngest_run_id,
      intent, sub_type, document_reference, language, confidence, urgency,
      intent_version, reasoning, body_version, detected_tone,
      tool_outputs, draft_url, status,
      human_verdict, human_notes, verdict_set_at, verdict_set_by,
      jsonb_build_object('entity', entity),
      created_at, updated_at, completed_at
    from debtor.agent_runs
    on conflict (id) do nothing;
  end if;
end$$;

-- 4. Drop Realtime pub entry + trigger + old table.
do $$
begin
  if exists (
    select 1 from pg_publication_tables
    where pubname='supabase_realtime' and schemaname='debtor' and tablename='agent_runs'
  ) then
    alter publication supabase_realtime drop table debtor.agent_runs;
  end if;
end$$;

drop trigger if exists agent_runs_set_updated_at on debtor.agent_runs;
drop function if exists debtor.agent_runs_set_updated_at();
drop table if exists debtor.agent_runs cascade;

-- 5. Indexes on public.agent_runs.
create index if not exists agent_runs_email_created_idx
  on public.agent_runs (email_id, created_at desc);
create index if not exists agent_runs_swarm_type_idx
  on public.agent_runs (swarm_type, created_at desc);
create index if not exists agent_runs_needs_verdict_idx
  on public.agent_runs (created_at desc)
  where human_verdict is null;
create index if not exists agent_runs_versions_idx
  on public.agent_runs (intent_version, body_version)
  where intent_version is not null;
create index if not exists agent_runs_tool_outputs_gin_idx
  on public.agent_runs using gin (tool_outputs);
create index if not exists agent_runs_context_gin_idx
  on public.agent_runs using gin (context);

-- 6. updated_at trigger (public schema function).
create or replace function public.agent_runs_set_updated_at()
returns trigger language plpgsql as $$
begin new.updated_at := now(); return new; end;
$$;
drop trigger if exists agent_runs_set_updated_at on public.agent_runs;
create trigger agent_runs_set_updated_at
  before update on public.agent_runs
  for each row execute function public.agent_runs_set_updated_at();

-- 7. RLS + Realtime publication.
alter table public.agent_runs enable row level security;
do $$
begin
  if not exists (
    select 1 from pg_publication_tables
    where pubname='supabase_realtime' and schemaname='public' and tablename='agent_runs'
  ) then
    alter publication supabase_realtime add table public.agent_runs;
  end if;
end$$;
