    Run via supabase mcp `execute_sql`:

    **Latest probe run, per-mailbox coverage:**
    ```sql
    WITH latest AS (
      SELECT probe_run_id FROM pipeline_health
      ORDER BY recorded_at DESC LIMIT 1
    )
    SELECT
      ph.mailbox,
      ph.swarm_type,
      ph.stage1_count,
      ph.stage0_count,
      round(ph.coverage_pct * 100, 1) AS coverage_pct,
      ph.breached
    FROM pipeline_health ph
    JOIN latest USING (probe_run_id)
    ORDER BY ph.mailbox;
    ```

    SELECT mb.mailbox, mb.swarm_type, c.stage1_count, c.stage0_count,

    CROSS JOIN LATERAL public.stage0_coverage_24h(mb.mailbox, mb.swarm_type) c;
