       ```sql
       SELECT id, status, topic, result->>'stage', result->>'verdict', completed_at
       FROM automation_runs
       WHERE result->>'message_id' = '{messageId}'
         AND swarm_type IN ('debtor-email', 'sales-email')
       ORDER BY completed_at;

       -- C. pipeline_events has BOTH Stage 0 and Stage 1 rows for the resolved email_id
       SELECT stage, decision, decision_details->>'emit_source', created_at
       FROM pipeline_events
       WHERE email_id = '{resolvedEmailId}'
       ORDER BY stage, created_at;
       ```

    - email_pipeline.emails: 1 row
    - automation_runs: at least 2 rows — Stage 0 placeholder (status flipped to 'completed' or 'predicted'), plus the Stage 1 worker's dispatch outcome (predicted/completed/skipped depending on category)
    - pipeline_events: at least 2 rows — stage=0 (decision='safe', emit_source='main-path') AND stage=1 (decision matching the regex outcome)

       ```sql
       WHERE swarm_type = 'debtor-email' AND enabled = true
       LIMIT 5;
       ```
