/**
 * Phase 999.5 Fix — sibling helper to invokeOrqModel that hits the Orq
 * Deployments product (separate from Agents product; no shared queue).
 * Studio versions per-call config (model, prompt, JSON schema, fallback chain),
 * and the response carries per-call cost (verified Wave 0 smoke S-1; path
 * locked in `wave-0-orq-deployments-smoke.md`).
 *
 * Same OrqClientTimeoutError + 45s AbortController as invokeOrqModel.
 * Reuses loadAgentByKeyCached (60s TTL).
 *
 * Throws on:
 *   - agent not in registry / disabled
 *   - registry row missing deployment_key
 *   - ORQ_API_KEY missing
 *   - timeout (> CLIENT_DEADLINE_MS) → OrqClientTimeoutError
 *   - non-2xx response
 *   - JSON parse failure on choices[0].message.content
 */
export async function invokeOrqDeployment(
  agent_key: string,
  inputs: Record<string, unknown>,
  opts?: { metadata?: Record<string, unknown>; extraParams?: Record<string, unknown> },
): Promise<InvokeResult> {
  if (!ORQ_API_KEY) throw new Error("ORQ_API_KEY is not set");

  const agent = await loadAgentByKeyCached(agent_key);
  const deployment_key = agent.deployment_key;
  if (!deployment_key) {
    throw new Error(
      `Orq agent ${agent_key} registry row missing deployment_key — populate before invoking via Deployments-direct path`,
    );
  }

  // <SHAPE_FROM_SUMMARY>: per 999.5-01-SUMMARY.md "Studio template variable",
  // either pass `inputs` verbatim (discrete-vars) OR wrap as { json_input: JSON.stringify(inputs) }.
  // Hard-code the choice; do NOT branch at runtime.
  const inputsForRequest = <SHAPE_FROM_SUMMARY>;

  const requestBody = {
    key: deployment_key,
    inputs: inputsForRequest,
    stream: false,                                     // Pitfall 7: explicit
    ...(opts?.metadata ? { metadata: opts.metadata } : {}),
    ...(opts?.extraParams ? { extra_params: opts.extraParams } : {}),
  };

  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), CLIENT_DEADLINE_MS);
  let res: Response;
  try {
    res = await fetch(DEPLOYMENTS_INVOKE_URL, {
      method: "POST",
      headers: {
        "content-type": "application/json",
        authorization: `Bearer ${ORQ_API_KEY}`,
      },
      body: JSON.stringify(requestBody),
      signal: ctrl.signal,
    });
  } catch (err) {
    if ((err as { name?: string } | undefined)?.name === "AbortError") {
      throw new OrqClientTimeoutError(
        `Orq deployments deadline exceeded for ${agent_key} after ${CLIENT_DEADLINE_MS}ms`,
      );
    }
    throw err;
  } finally {
    clearTimeout(timer);
  }

  if (!res.ok) {
    const text = await res.text().catch(() => "");
    throw new Error(
      `Orq deployments invoke ${agent_key} failed: HTTP ${res.status} ${text.slice(0, 200)}`,
    );
  }

  const json = (await res.json()) as {
    id?: string;
    choices?: Array<{ message?: { content?: string | null } }>;
    usage?: { prompt_tokens?: number; completion_tokens?: number; total_tokens?: number };
    telemetry?: { traceId?: string; spanId?: string };
    // <LOCKED_PATH from Wave 0 smoke> — pin the type narrowly to the actual path.
  };

  const rawText = json.choices?.[0]?.message?.content ?? "";
  if (!rawText) {
    throw new Error(`Orq deployments invoke ${agent_key}: empty content in response`);
  }
  const stripped = rawText.trim()
    .replace(/^```(?:json)?\s*/i, "")
    .replace(/\s*```$/i, "");
  let raw: unknown;
  try {
    raw = JSON.parse(stripped);
  } catch (e) {
    throw new Error(
      `Orq deployments invoke ${agent_key}: JSON.parse failed: ${(e as Error).message}`,
    );
  }

  // <LOCKED_PATH>: read EXACTLY the path from wave-0-orq-deployments-smoke.md §A1.
  // Example shapes (pick ONE based on smoke evidence — do NOT include a fallthrough):
  //   const total_cost = json.billing?.total_cost ?? 0;
  //   const total_cost = json.cost ?? 0;
  //   const total_cost = (json as { providerResponse?: { usage?: { total_cost?: number } } })
  //     .providerResponse?.usage?.total_cost ?? 0;
  const total_cost = <READ_FROM_LOCKED_PATH> ?? 0;
  const cost_cents = Math.round(total_cost * 100);

  return {
    raw,
    agent,
    usage: {
      prompt_tokens: json.usage?.prompt_tokens ?? 0,
      completion_tokens: json.usage?.completion_tokens ?? 0,
      total_tokens: json.usage?.total_tokens ?? 0,
    },
    billing: { total_cost },
    cost_cents,
  };
}
