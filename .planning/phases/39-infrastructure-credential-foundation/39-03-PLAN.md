```typescript
import type { HealthUpdatePayload } from "@/lib/credentials/types";

/**
 * Broadcast health check results to the health dashboard.
 * Called from Inngest health-check function after each service check.
 */
export async function broadcastHealthUpdate(
  payload: HealthUpdatePayload
): Promise<void> {
  const admin = createAdminClient();
  const channel = admin.channel("health:status");
  await channel.send({
    type: "broadcast",
    event: "health-update",
    payload,
  });
  admin.removeChannel(channel);
}
```

**File 2: web/app/api/health-check/route.ts**

Create API route for triggering and retrieving health checks.

POST handler (trigger health check):
- Authenticate via createClient() then getUser()
- Return 401 if not authenticated
- Send Inngest event: `inngest.send({ name: "infrastructure/health-check.requested", data: { requestedBy: user.id } })`
- Return 200 with `{ triggered: true }`

GET handler (get latest results):
- Authenticate via createClient() then getUser()
- Return 401 if not authenticated
- Query health_checks table via admin client (RLS allows all authenticated users)
- SELECT all columns WHERE id = 'latest'
- Return the health check result or null if none exists

```typescript
import { NextRequest, NextResponse } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { createAdminClient } from "@/lib/supabase/admin";
import { inngest } from "@/lib/inngest/client";

export async function POST(req: NextRequest) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  await inngest.send({
    name: "infrastructure/health-check.requested",
    data: { requestedBy: user.id },
  });

  return NextResponse.json({ triggered: true });
}

export async function GET(req: NextRequest) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const admin = createAdminClient();
  const { data } = await admin
    .from("health_checks")
    .select("*")
    .eq("id", "latest")
    .single();

  return NextResponse.json({ result: data || null });
}
```
