**B) `use-counts-rpc.ts`** (client hook) — copy broadcast-subscribe pattern from `automation-realtime-provider.tsx` lines 58-80:

```typescript
"use client";
import { useEffect, useState } from "react";
import { createClient } from "@/lib/supabase/client";

export type LabelCount = { nxt_database: string; source_mailbox: string; method: string; status: string; confidence: string; total: number };
export type LabelRow = { /* matching the page.tsx select shape */ id: string; graph_message_id: string; /* etc */ [key: string]: unknown };

export function useLabelingRealtime(initial: { rows: LabelRow[]; counts: LabelCount[] }) {
  const [rows, setRows] = useState(initial.rows);
  const [counts, setCounts] = useState(initial.counts);

  useEffect(() => {
    const supabase = createClient();
    let cancelled = false;
    const refetch = async () => {
      // Re-fetch counts + rows via Next.js Route Handler (or direct query).
      // For first cut, refresh page via location.reload() OR use a /api/automations/debtor-email-labeling/data route.
      const res = await fetch("/api/automations/debtor-email-labeling/data");
      if (cancelled || !res.ok) return;
      const j = await res.json();
      setRows(j.rows); setCounts(j.counts);
    };
    const channel = supabase.channel("automations:debtor-email-labeling:stale");
    channel.on("broadcast", { event: "stale" }, () => refetch()).subscribe();
    return () => { cancelled = true; supabase.removeChannel(channel); };
  }, []);

  return { rows, counts };
}
```

```typescript
"use server";
import { createAdminClient } from "@/lib/supabase/admin";
import { emitAutomationRunStale } from "@/lib/automations/runs/emit";
import { revalidatePath } from "next/cache";

export async function approveLabel(args: { agent_run_id: string; email_label_id: string; reviewer: string }) {
  const admin = createAdminClient();
  const now = new Date().toISOString();
  await admin.from("agent_runs").update({
    human_verdict: "approve", verdict_set_at: now,
  }).eq("id", args.agent_run_id);
  await admin.schema("debtor").from("email_labels").update({
    reviewed_by: args.reviewer, reviewed_at: now,
  }).eq("id", args.email_label_id);
  await emitAutomationRunStale(admin, "debtor-email-labeling");
  revalidatePath("/automations/debtor-email-labeling");
}

export async function rejectLabel(args: {
  agent_run_id: string; email_label_id: string; reviewer: string;
  corrected_account_id: string | null;
}) {
  const admin = createAdminClient();
  const now = new Date().toISOString();
  await admin.from("agent_runs").update({
    human_verdict: "reject", verdict_set_at: now,
    corrected_category: args.corrected_account_id,
  }).eq("id", args.agent_run_id);
  await admin.schema("debtor").from("email_labels").update({
    reviewed_by: args.reviewer, reviewed_at: now,
  }).eq("id", args.email_label_id);
  await emitAutomationRunStale(admin, "debtor-email-labeling");
  revalidatePath("/automations/debtor-email-labeling");
}
```

```typescript
"use client";
import { useState } from "react";
import { useLabelingRealtime, type LabelRow, type LabelCount } from "./use-counts-rpc";
import { LabelingRowDrawer } from "./labeling-row-drawer";

export function LabelingRowList({ rows: initialRows, counts: initialCounts }: { rows: LabelRow[]; counts: LabelCount[] }) {
  const { rows, counts } = useLabelingRealtime({ rows: initialRows, counts: initialCounts });
  const [openRow, setOpenRow] = useState<LabelRow | null>(null);

  // Group by source_mailbox
  const byMailbox = new Map<string, LabelRow[]>();
  for (const r of rows) {
    const k = (r.source_mailbox as string) ?? "(no mailbox)";
    if (!byMailbox.has(k)) byMailbox.set(k, []);
    byMailbox.get(k)!.push(r);
  }

  return (
    <div className="mt-8 space-y-8">
      {/* counts grid */}
      <section className="grid grid-cols-3 gap-4">
        {counts.map(c => (
          <div key={`${c.nxt_database}-${c.method}-${c.status}`} className="px-4 py-3 rounded-[var(--v7-radius-card)] bg-[var(--v7-bg-subtle)]">
            <div className="text-[12px] text-[var(--v7-fg-muted)]">{c.nxt_database || "—"} · {c.method} · {c.status}</div>
            <div className="text-[20px] font-semibold">{c.total}</div>
          </div>
        ))}
      </section>

      {/* rows by mailbox */}
      {Array.from(byMailbox.entries()).map(([mailbox, rs]) => (
        <section key={mailbox}>
          <h2 className="text-[16px] font-semibold mb-3">{mailbox}</h2>
          <table className="w-full text-[13px]">
            <thead><tr className="text-left text-[var(--v7-fg-muted)]"><th>Subject</th><th>Sender</th><th>Account</th><th>Method</th><th>Confidence</th><th>Status</th></tr></thead>
            <tbody>
              {rs.map(r => (
                <tr key={r.id as string} className="hover:bg-[var(--v7-bg-subtle)] cursor-pointer" onClick={() => setOpenRow(r)}>
                  <td className="py-2 pr-3 truncate max-w-[280px]">{(r.subject as string) ?? r.graph_message_id}</td>
                  <td className="py-2 pr-3">{(r.from_email as string) ?? "—"}</td>
                  <td className="py-2 pr-3">{(r.customer_account_id as string) ?? "—"}</td>
                  <td className="py-2 pr-3">{r.method as string}</td>
                  <td className="py-2 pr-3">{r.confidence as string}</td>
                  <td className="py-2 pr-3">{r.status as string}</td>
                </tr>
              ))}
            </tbody>
          </table>
        </section>
      ))}

      <LabelingRowDrawer row={openRow} onClose={() => setOpenRow(null)} />
    </div>
  );
}
```

```typescript
import { useState, useTransition } from "react";
import { Sheet, SheetContent, SheetTitle, SheetDescription, SheetClose } from "@/components/ui/sheet";

      <div className="text-[14px] font-semibold">{value || "—"}</div>

  );
}
```
