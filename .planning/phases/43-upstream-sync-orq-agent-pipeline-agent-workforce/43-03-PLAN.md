**Create `web/lib/pipeline/systems-serializer.ts`:**

```typescript
/**
 * Serializes DB system records to markdown matching the upstream systems.md format.
 *
 * The architect agent expects systems context in the same format as the CLI's
 * systems.md file. This function converts database records to that format.
 */

import type { System } from "@/lib/systems/types";

export function serializeSystemsToMarkdown(systems: System[]): string {
  if (systems.length === 0) return "";

  const lines = ["# Systems Registry", ""];
  for (const sys of systems) {
    lines.push(`### ${sys.name}`);
    lines.push(`- **Integration method:** ${sys.integration_method}`);
    if (sys.url) lines.push(`- **URL:** ${sys.url}`);
    if (sys.auth_notes) lines.push(`- **Auth:** ${sys.auth_notes}`);
    if (sys.notes) lines.push(`- **Notes:** ${sys.notes}`);
    lines.push("");
  }
  return lines.join("\n");
}
```

```typescript
// Step: Fetch project-linked systems for architect context
const systemsContext = await step.run("fetch-systems-context", async () => {
  const admin = createAdminClient();

  // Get project ID from the run
  const { data: run } = await admin
    .from("pipeline_runs")
    .select("project_id")
    .eq("id", runId)
    .single();

  if (!run?.project_id) return { markdown: "" };

  // Get linked system IDs for this project
  const { data: links } = await admin
    .from("system_project_links")
    .select("system_id")
    .eq("project_id", run.project_id);

  if (!links || links.length === 0) return { markdown: "" };

  // Fetch full system records
  const systemIds = links.map(l => l.system_id);
  const { data: systems } = await admin
    .from("systems")
    .select("id, name, integration_method, url, auth_notes, notes, created_by, created_at, updated_at")
    .in("id", systemIds);

  if (!systems || systems.length === 0) return { markdown: "" };

  return { markdown: serializeSystemsToMarkdown(systems as import("@/lib/systems/types").System[]) };
});
```
