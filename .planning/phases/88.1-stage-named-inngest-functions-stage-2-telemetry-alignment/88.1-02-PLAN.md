```typescript
import { describe, expect, it } from "vitest";
import { execSync } from "node:child_process";
import path from "node:path";

const LEGACY_STRINGS = [
  "classifier/label-resolver",
  "automations/debtor-email-icontroller-tagger",
  "automations/debtor-email-icontroller-shard-worker",
  "automations/debtor-email-icontroller-dispatch",
  "debtor-email/label-resolve.requested",
  "debtor-email/icontroller-tag.requested",
  "icontroller/cleanup.shard.requested",
] as const;

// Path to web/ from the test file location.
const WEB_ROOT = path.resolve(__dirname, "../..");
const SELF = path.relative(WEB_ROOT, __filename);

function grepLegacy(needle: string): string[] {
  // ripgrep is on the dev machines + CI; if absent the test errors loud,
  // which is fine — we want to know.
  try {
    const out = execSync(
      `rg -nF --type-add 'src:*.{ts,tsx,md}' -tsrc -g '!node_modules' -g '!.next' -g '!dist' -g '!coverage' -g '!${SELF}' -- ${JSON.stringify(needle)} .`,
      { cwd: WEB_ROOT, encoding: "utf8" },
    );
    return out
      .split("\n")
      .filter((line) => line.length > 0)
      .map((line) => line.trim());
  } catch (err) {
    // rg exits 1 when no matches — that's the GREEN state.
    const e = err as { status?: number; stdout?: string };
    if (e.status === 1) return [];
    throw err;
  }
}

describe("no legacy Inngest names", () => {
  it("web/ is free of pre-Phase-88.1 function ids and event names", () => {
    const offenders: Array<{ needle: string; hit: string }> = [];
    for (const needle of LEGACY_STRINGS) {
      const hits = grepLegacy(needle);
      for (const hit of hits) offenders.push({ needle, hit });
    }
    if (offenders.length > 0) {
      const summary = offenders
        .map(({ needle, hit }) => `  [${needle}] ${hit}`)
        .join("\n");
      throw new Error(
        `Found ${offenders.length} legacy Inngest name occurrence(s) under web/. ` +
          `These MUST be renamed per Phase 88.1:\n${summary}`,
      );
    }
    expect(offenders.length).toBe(0);
  });

  // Defense-in-depth (Phase 65 lesson). `const send = inngest.send` loses
  // the `this`-binding and the first call throws TypeError. Any rename
  // task that touches a producer site must keep the call INLINE.
  it("no destructured inngest.send anywhere under web/", () => {
    const cmd =
      `rg -nE 'const\\s+send\\s*=\\s*inngest\\.send' web/ ` +
      `--type=ts --type=tsx ` +
      `-g '!**/node_modules/**' -g '!**/.next/**' ` +
      `|| true`;
    const out = execSync(cmd, { cwd: REPO_ROOT, encoding: "utf8" }).trim();
    if (out.length > 0) {
      throw new Error(
        `Found destructured inngest.send (loses this-binding — see Phase 65 lesson):\n${out}`,
      );
    }
    expect(out.length).toBe(0);
  });
});
```
