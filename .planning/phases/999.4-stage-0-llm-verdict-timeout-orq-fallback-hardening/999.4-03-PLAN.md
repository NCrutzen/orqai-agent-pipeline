```typescript
type PromptCacheEntry = { prompt: string; fetched_at: number };
const promptCache = new Map<string, PromptCacheEntry>();

async function loadSystemPromptForAgent(orqai_id: string): Promise<string> {
  const now = Date.now();
  const hit = promptCache.get(orqai_id);
  if (hit && now - hit.fetched_at < SYSTEM_PROMPT_CACHE_TTL_MS) return hit.prompt;

  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), CLIENT_DEADLINE_MS);
  try {
    const res = await fetch(`https://api.orq.ai/v2/agents/${orqai_id}`, {
      method: "GET",
      headers: { authorization: `Bearer ${ORQ_API_KEY}` },
      signal: ctrl.signal,
    });
    if (!res.ok) {
      const text = await res.text().catch(() => "");
      throw new Error(
        `Orq get_agent ${orqai_id} failed: HTTP ${res.status} ${text.slice(0, 200)}`,
      );
    }
    const json = (await res.json()) as {
      // Orq agent manifest — system prompt location depends on Studio version.
      // Verify against the Wave 0 smoke evidence and CLAUDE.md §Orq.ai docs.
      // Common shapes: { prompts: { system: string } } OR { system_prompt: string }
      // OR { messages: [{ role: "system", content: string }] }.
      // Implement defensively — try each, throw if none present.
      prompts?: { system?: string };
      system_prompt?: string;
      messages?: Array<{ role: string; content: string }>;
    };
    const prompt =
      json.prompts?.system
      ?? json.system_prompt
      ?? json.messages?.find((m) => m.role === "system")?.content;
    if (!prompt) {
      throw new Error(
        `Orq get_agent ${orqai_id}: no system prompt found in agent manifest (shape may have changed; check Studio)`,
      );
    }
    promptCache.set(orqai_id, { prompt, fetched_at: now });
    return prompt;
  } catch (err) {
    if ((err as { name?: string } | undefined)?.name === "AbortError") {
      throw new OrqClientTimeoutError(
        `Orq get_agent deadline exceeded for ${orqai_id} after ${CLIENT_DEADLINE_MS}ms`,
      );
    }
    throw err;
  } finally {
    clearTimeout(timer);
  }
}
```

3. **Sibling helper `invokeOrqModel`** (exported, after `invokeOrqAgent`):

```typescript
export async function invokeOrqModel(
  agent_key: string,
  inputs: Record<string, unknown>,
): Promise<InvokeResult> {
  if (!ORQ_API_KEY) throw new Error("ORQ_API_KEY is not set");
  const agent = await loadAgent(agent_key);
  if (!agent.enabled) {
    throw new Error(`Orq agent ${agent_key} is not enabled in registry`);
  }
  const systemPrompt = await loadSystemPromptForAgent(agent.orqai_id);

  const mc = (agent.model_config ?? {}) as {
    primary?: string;
    fallbacks?: string[];
    temperature?: number;
    max_tokens?: number;
  };
  const model = mc.primary;
  if (!model) {
    throw new Error(`Orq agent ${agent_key} registry row missing model_config.primary`);
  }

  const requestBody = {
    model,
    fallback_models: mc.fallbacks ?? [],
    temperature: mc.temperature ?? 0,
    max_tokens: mc.max_tokens ?? 600,
    messages: [
      { role: "system" as const, content: systemPrompt },
      { role: "user" as const, content: JSON.stringify(inputs, null, 2) },
    ],
    response_format: {
      type: "json_schema" as const,
      json_schema: {
        name: agent.agent_key.replace(/-/g, "_"),
        strict: true as const,
        schema: agent.output_schema,
      },
    },
  };

  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), CLIENT_DEADLINE_MS);
  let res: Response;
  try {
    res = await fetch(ROUTER_URL, {
      method: "POST",
      headers: {
        "content-type": "application/json",
        authorization: `Bearer ${ORQ_API_KEY}`,
      },
      body: JSON.stringify(requestBody),
      signal: ctrl.signal,
    });
  } catch (err) {
    if ((err as { name?: string } | undefined)?.name === "AbortError") {
      throw new OrqClientTimeoutError(
        `Orq router deadline exceeded for ${agent_key} after ${CLIENT_DEADLINE_MS}ms`,
      );
    }
    throw err;
  } finally {
    clearTimeout(timer);
  }

  if (!res.ok) {
    const text = await res.text().catch(() => "");
    throw new Error(
      `Orq router invoke ${agent_key} failed: HTTP ${res.status} ${text.slice(0, 200)}`,
    );
  }
  const json = (await res.json()) as {
    choices?: Array<{ message?: { content?: string } }>;
    usage?: { prompt_tokens?: number; completion_tokens?: number; total_tokens?: number };
  };
  const rawText = json.choices?.[0]?.message?.content ?? "";
  if (!rawText) {
    throw new Error(`Orq router invoke ${agent_key}: empty content in response`);
  }
  const stripped = rawText
    .trim()
    .replace(/^```(?:json)?\s*/i, "")
    .replace(/\s*```$/i, "");
  const raw = JSON.parse(stripped);

  return {
    raw,
    agent,
    usage: {
      prompt_tokens: json.usage?.prompt_tokens ?? 0,
      completion_tokens: json.usage?.completion_tokens ?? 0,
      total_tokens: json.usage?.total_tokens ?? 0,
    },
    billing: { total_cost: 0 }, // Router-path billing not yet surfaced in response; fill in when Orq exposes it
    cost_cents: 0,
  };
}
```
