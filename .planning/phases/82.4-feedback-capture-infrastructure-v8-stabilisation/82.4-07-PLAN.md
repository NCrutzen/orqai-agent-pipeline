```typescript
import { inngest } from "@/lib/inngest/client";
import { createAdminClient } from "@/lib/supabase/admin";

const BUCKET = "email-feedback-snapshots";
// Window covers 26 hours so consecutive nightly runs overlap slightly —
// guards against clock skew between Inngest scheduler and Supabase clock.
const WINDOW_HOURS = 26;

export const emailFeedbackSnapshot = inngest.createFunction(
  { id: "feedback/nightly-snapshot", retries: 1 },
  { cron: "TZ=Europe/Amsterdam 0 2 * * *" },
  async ({ step }) => {
    // Phase 65 replay-id rule: mint inside step.run so a replayed tick
    // groups under the same id rather than minting a fresh one.
    const run_id = await step.run("resolve-run-id", async () => crypto.randomUUID());

    const snapshot_date = await step.run("resolve-snapshot-date", async () => {
      const d = new Date();
      const yyyy = d.getUTCFullYear();
      const mm = String(d.getUTCMonth() + 1).padStart(2, "0");
      const dd = String(d.getUTCDate()).padStart(2, "0");
      return `${yyyy}-${mm}-${dd}`;
    });

    const rows = await step.run("read-feedback-window", async () => {
      const admin = createAdminClient();
      const since = new Date(Date.now() - WINDOW_HOURS * 60 * 60 * 1000).toISOString();
      const { data, error } = await admin
        .from("email_feedback")
        .select("id, email_id, stage, verdict, corrected_value, prose_notes, operator_id, created_at")
        .gte("created_at", since)
        .order("created_at", { ascending: true })
        .limit(10000);
      if (error) throw new Error(`email_feedback read failed: ${error.message}`);
      return data ?? [];
    });

    const storage_path = await step.run("upload-snapshot", async () => {
      const admin = createAdminClient();
      const path = `${BUCKET === "email-feedback-snapshots" ? "" : ""}${snapshot_date}.json`;
      // Path inside bucket is just the filename; upload() resolves relative to the bucket
      const buffer = Buffer.from(JSON.stringify(rows, null, 2));
      const { error } = await admin.storage
        .from(BUCKET)
        .upload(`${snapshot_date}.json`, buffer, {
          contentType: "application/json",
          upsert: true,
        });
      if (error) throw new Error(`Storage upload failed: ${error.message}`);
      return `${BUCKET}/${snapshot_date}.json`;
    });

    if (rows.length >= 10000) {
      console.warn(`[email-feedback-snapshot] row cap reached (10000) on run ${run_id} — increase WINDOW_HOURS clamp or page through.`);
    }

    return { run_id, snapshot_date, row_count: rows.length, storage_path };
  },
);
```
