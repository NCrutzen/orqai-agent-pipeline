```typescript
import type { SupabaseClient } from "@supabase/supabase-js";

export interface FeedbackListParams {
  stage: 0 | 1 | 2 | 3;
  swarmType: string;
  /** ISO timestamp cursor: only return rows with received_at < before. Omit for first page. */
  before?: string;
  /** When true, restrict to rows where the stage decision is in a "needs action" state. */
  needsActionOnly?: boolean;
  /** When true, restrict to emails where the current operator has at least one email_feedback row. */
  mineOnly?: boolean;
  /** Server-stamped operator_id; required when mineOnly=true. */
  operatorId?: string;
  /** Page size. Default 50. Cap 100. */
  limit?: number;
}

export interface FeedbackListRow {
  email_id: string;
  pipeline_event_id: string;        // Stage-N pipeline_events.id
  stage: 0 | 1 | 2 | 3;
  stage_state: string;              // pipeline_events.decision or stage_state column
  // From email_pipeline.emails:
  subject: string | null;
  sender_email: string | null;
  sender_name: string | null;
  received_at: string;              // ISO; used as the cursor
  mailbox_id: number | null;
  // Sort bucket — derived in the loader:
  sort_bucket: "needs_action" | "auto_handled" | "own_reviewed";
  // Latest own feedback (when mineOnly OR opportunistic join), null otherwise:
  own_latest_verdict: "confirm" | "override" | "unclear" | null;
  own_latest_at: string | null;
}

export interface FeedbackListPage {
  rows: FeedbackListRow[];
  /** ISO timestamp to feed back as `before` for next page; null when no more rows. */
  nextBefore: string | null;
}

export async function loadStageFeedbackList(
  admin: SupabaseClient,
  params: FeedbackListParams,
): Promise<FeedbackListPage>;
```

Query strategy (3 queries, joined in JS, mirrors kanban-loader pattern):
1. Query `pipeline_events`: `select id, email_id, stage, decision, created_at` where `swarm_type=$swarm AND stage=$stage`. Order `created_at desc`. Limit `limit*2` to give bucket-sort room. When `before` set, filter `created_at < before` (use pipeline_events.created_at as the cursor proxy; final cursor returned to caller is the JOINED received_at for the last row in the returned page).
   - When `needsActionOnly`: add `.in("decision", [needs_action variants])` — list TBD by reading existing 82.3 logic but at minimum the values that produce orange-circle rows. If unsure, use `.eq("decision", "needs_review")` plus any other value that the existing chips identify as needs-action; grep `stage_state.*needs` in `web/app/(dashboard)/automations/[swarm]/` for the canonical set.
2. Query `email_pipeline.emails` by `id IN (<unique email_ids from query 1>)`: fetch `id, subject, sender_email, sender_name, received_at, mailbox_id`. Build a Map.
3. Query `email_feedback` by `email_id IN (<set>) AND stage=$stage`. If `operatorId` provided, also filter `operator_id=$operatorId`. Order `created_at desc`. Build a Map keyed by `email_id` taking first (latest) per email.

Merge:
- Drop rows where the email_metadata Map lookup is null (email row was deleted).
- Compute `sort_bucket`:
  - `needs_action` if `pipeline_events.decision` is in the needs-action set
  - `own_reviewed` if `own_latest_verdict !== null` AND not needs-action
  - `auto_handled` otherwise
- Sort: bucket priority (needs_action < auto_handled < own_reviewed) then `received_at desc`.
- Apply `mineOnly` filter AFTER merge: keep only rows where `own_latest_verdict !== null`.
- Truncate to `limit`.
- `nextBefore = rows.length === limit ? rows[rows.length-1].pipeline_event_created_at : null` — MUST be the same column used for the `<` filter to avoid skipped/duplicate rows across pages. Carry `pipeline_events.created_at` through the merge as an internal field on each row (does not need to surface in `FeedbackListRow` if undesired, but the loader must return the cursor value from this column).

Sales-email Stage 0/1 tabs: same loader, same shape, swarmType='sales-email'. Stage 2/3 sales-email pages render the existing empty-state UI separately (no loader change needed).
