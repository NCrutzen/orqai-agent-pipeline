// Activity time series
const activityData = (historicalResult.data ?? []).map((s: any) => {
  const m = s.metrics;
  return {
    date: s.computed_at.slice(0, 10),
    pipeline: m?.runsBySource?.pipeline ?? 0,
    zapier: m?.runsBySource?.zapier ?? 0,
    orqai: m?.runsBySource?.orqai ?? 0,
  };
});

// Success rate time series
const successRateData = (historicalResult.data ?? []).map((s: any) => ({

}));

// Cost per run time series
const costTrendData = (historicalResult.data ?? []).map((s: any) => ({

}));
