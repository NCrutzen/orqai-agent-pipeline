```typescript
// Phase 65 (D-09) — pure tri-state escalation gate (CORD-02).
// Order: low_confidence FIRST, then high_intent_count, then requires_orchestration_flag.

import type { IntentAgentOutputV2 } from "../triage/types";
import type { SwarmCategoryRow } from "@/lib/swarms/types";

export type EscalationDecision =
  | { kind: "single_shot" }
  | { kind: "orchestrator"; reason: "low_confidence" | "high_intent_count" | "requires_orchestration_flag" };

export function evaluateEscalationGate(
  output: IntentAgentOutputV2,
  categories: SwarmCategoryRow[],
): EscalationDecision {
  if (output.ranked[0].confidence === "low") {
    return { kind: "orchestrator", reason: "low_confidence" };
  }
  if (output.ranked.length >= 3) {
    return { kind: "orchestrator", reason: "high_intent_count" };
  }
  const flagged = output.ranked.some(r =>
    categories.find(c => c.category_key === r.intent)?.requires_orchestration === true
  );
  if (flagged) {
    return { kind: "orchestrator", reason: "requires_orchestration_flag" };
  }
  return { kind: "single_shot" };
}
```

```typescript
import { inngest } from "@/lib/inngest/client";
import { createAdminClient } from "@/lib/supabase/admin";
import { findCachedOutput, mergeToolOutputs, updateRun } from "@/lib/automations/debtor-email/triage/agent-runs";
import { invokeIntentAgent } from "@/lib/automations/debtor-email/triage/invoke-intent";
import { INTENT_VERSION_V2 } from "@/lib/automations/debtor-email/triage/types";
import { loadSwarmCategories } from "@/lib/swarms/registry";
import { evaluateEscalationGate } from "@/lib/automations/debtor-email/coordinator/escalation-gate";
import { emitAutomationRunStale } from "@/lib/automations/runs/emit";

export const debtorEmailTriage = inngest.createFunction(
  {
    id: "automations/debtor-email-triage",
    retries: 0,
    concurrency: [
      { key: "event.data.entity", limit: 4 },
      { key: "event.data.run_id", limit: 1 },
    ],
  },
  { event: "debtor/email.received" }, // verify exact event name from existing function
  async ({ event, step }) => {
    const admin = createAdminClient();
    const { run_id, email_id, automation_run_id, swarm_type = "debtor-email", budget_run_id, agent_run_id, /* etc */ } = event.data;

    // 1. Insert coordinator_runs row early so the row exists for the escalation update step.
    await step.run("create-coordinator-run", async () => {
      await admin.from("coordinator_runs").insert({
        run_id,
        automation_run_id,
        email_id,
        swarm_type,
        ranked_intents: [],            // populated after classify
        escalation_decision: "single_shot", // tentative, overwritten below
        expected_handlers: 1,
        budget_run_id,
      });
      await emitAutomationRunStale(admin, "debtor-email-review");
    });

    try {
      // 2. Classify ranked output, with idempotent cache via INTENT_VERSION_V2 literal.
      const output = await step.run("classify-intent", async () => {
        const cached = await findCachedOutput(admin, email_id, "intent_version", INTENT_VERSION_V2, "tool_outputs");
        const cachedFirst = cached?.intent_first_pass;
        if (cachedFirst) return cachedFirst;

        const { output: fresh } = await invokeIntentAgent({ /* same input args as v1 */ });
        await mergeToolOutputs(admin, agent_run_id, "intent_first_pass", fresh);
        await updateRun(admin, agent_run_id, {
          intent_version: INTENT_VERSION_V2,
          // top-1 hoisted for back-compat columns:
          intent: fresh.ranked[0].intent,
          confidence: fresh.ranked[0].confidence,
          language: fresh.language,
          urgency: fresh.urgency,
        });
        return fresh;
      });

      // 3. Persist ranked_intents on coordinator_runs.
      await step.run("persist-ranked", async () => {
        await admin.from("coordinator_runs")
          .update({ ranked_intents: output.ranked })
          .eq("run_id", run_id);
      });

      // 4. Evaluate escalation gate.
      const decision = await step.run("evaluate-escalation-gate", async () => {
        const categories = await loadSwarmCategories(admin, "debtor-email");
        return evaluateEscalationGate(output, categories);
      });

      // 5. Update coordinator_runs.escalation_decision + reason.
      await step.run("write-escalation", async () => {
        await admin.from("coordinator_runs")
          .update({
            escalation_decision: decision.kind,
            escalation_reason: decision.kind === "orchestrator" ? decision.reason : null,
          })
          .eq("run_id", run_id);
      });

      if (decision.kind === "single_shot") {
        // 6a. Single-shot path (CORD-04): registry-driven dispatch — same idiom as classifier-verdict-worker:149-176.
        const top = output.ranked[0];
        const categories = await loadSwarmCategories(admin, "debtor-email");
        const category = categories.find(c => c.category_key === top.intent);
        if (!category?.swarm_dispatch) {
          throw new Error(`no swarm_dispatch registered for intent=${top.intent} (RESEARCH Pitfall 1 — verify Plan 01 seed migration applied)`);
        }
        await step.run("dispatch-single-shot", async () => {
          await (inngest.send as unknown as (p: { name: string; data: Record<string, unknown> }) => Promise<unknown>)({
            name: category.swarm_dispatch!,
            data: { run_id, email_id, automation_run_id, intent: top.intent, ranked: output.ranked, budget_run_id, swarm_type },
          });
        });
        await step.run("mark-coordinator-complete", async () => {
          await admin.from("coordinator_runs")
            .update({ completed_at: new Date().toISOString(), completed_handlers: 1 })
            .eq("run_id", run_id);
          await emitAutomationRunStale(admin, "debtor-email-review");
        });
      } else {
        // 6b. Orchestrator path: emit single event, Plan 04's orchestrator function takes over.
        await step.run("dispatch-orchestrator", async () => {
          await (inngest.send as unknown as (p: { name: string; data: Record<string, unknown> }) => Promise<unknown>)({
            name: "debtor-email/orchestrator.requested",
            data: { run_id, email_id, automation_run_id, ranked: output.ranked, language: output.language, urgency: output.urgency, escalation_reason: decision.reason, budget_run_id, swarm_type },
          });
        });
        // expected_handlers stays 1 here — Plan 04's orchestrator updates to N after planner runs.
      }
    } catch (err) {
      const msg = err instanceof Error ? err.message : String(err);
      await step.run("mark-failed", async () => {
        await admin.from("automation_runs").update({
          status: "failed",
          error_message: msg,
          completed_at: new Date().toISOString(),
        }).eq("id", automation_run_id);
        await admin.from("coordinator_runs").update({
          completed_at: new Date().toISOString(),
        }).eq("run_id", run_id);
        await emitAutomationRunStale(admin, "debtor-email-review");
      });
      throw err;
    }
  },
);
```
