```typescript
#!/usr/bin/env tsx
/**
 * Phase 65 regression backfill (RESEARCH OQ5 + CONTEXT Claude's Discretion).
 *
 * Re-runs the new v2 coordinator (debtor-intent-agent v2 with ranked output) over recent
 * production debtor-email rows, and compares top-1 v2 intent against the v1 intent already
 * stored in agent_runs.tool_outputs.intent_first_pass. Produces a markdown report that
 * confirms (or contradicts) the ~80% single-shot success criterion (CORD-04).
 *
 * Usage:
 *   cd /Users/nickcrutzen/Developer/agent-workforce
 *   tsx scripts/phase-65-regression-backfill.ts --limit 200 --days 14
 *
 * Acceptance gate: report shows >=70% top-1 agreement (sanity check) and an escalation
 * distribution where single_shot/total >=70% (loose lower bound; ~80% target lives in
 * the production rollout).
 */

import { createAdminClient } from "@/lib/supabase/admin";
import { invokeIntentAgent } from "@/lib/automations/debtor-email/triage/invoke-intent";
import { loadSwarmCategories } from "@/lib/swarms/registry";
import { evaluateEscalationGate } from "@/lib/automations/debtor-email/coordinator/escalation-gate";
import { writeFileSync } from "node:fs";
import { resolve } from "node:path";

const argv = process.argv.slice(2);
const arg = (k: string, def: string) => {
  const i = argv.indexOf(`--${k}`);
  return i >= 0 ? argv[i + 1] : def;
};
const LIMIT = parseInt(arg("limit", "200"), 10);
const DAYS = parseInt(arg("days", "14"), 10);

interface Comparison {
  email_id: string;
  agent_run_id: string;
  v1_intent: string | null;
  v1_confidence: string | null;
  v2_top1_intent: string;
  v2_top1_confidence: string;
  v2_ranked_count: number;
  agreement: boolean;
  escalation: { kind: string; reason?: string };
}

async function main() {
  const admin = createAdminClient();
  const since = new Date(Date.now() - DAYS * 86400_000).toISOString();

  const { data: rows, error } = await admin.from("agent_runs")
    .select("id, email_id, tool_outputs, language, urgency")
    .eq("swarm_type", "debtor-email")
    .gte("created_at", since)
    .order("created_at", { ascending: false })
    .limit(LIMIT);
  if (error) throw error;

  const categories = await loadSwarmCategories(admin, "debtor-email");
  const results: Comparison[] = [];

  for (const r of rows ?? []) {
    let toolOutputs = r.tool_outputs as unknown;
    while (typeof toolOutputs === "string") toolOutputs = JSON.parse(toolOutputs as string);
    const v1 = (toolOutputs as { intent_first_pass?: { intent?: string; confidence?: string } })?.intent_first_pass;
    if (!v1?.intent) continue; // skip rows without a v1 intent

    try {
      const { output } = await invokeIntentAgent({ /* shape matches Plan 03 invokeIntentAgent input — adapt to actual signature */ email_id: r.email_id });
      const decision = evaluateEscalationGate(output, categories);
      results.push({
        email_id: r.email_id,
        agent_run_id: r.id,
        v1_intent: v1.intent ?? null,
        v1_confidence: v1.confidence ?? null,
        v2_top1_intent: output.ranked[0].intent,
        v2_top1_confidence: output.ranked[0].confidence,
        v2_ranked_count: output.ranked.length,
        agreement: v1.intent === output.ranked[0].intent,
        escalation: { kind: decision.kind, reason: decision.kind === "orchestrator" ? decision.reason : undefined },
      });
    } catch (err) {
      console.warn(`skip ${r.email_id}: ${err instanceof Error ? err.message : String(err)}`);
    }
    // Throttle to avoid Orq rate limit (rough: 5/s).
    await new Promise(r => setTimeout(r, 200));
  }

  const total = results.length;
  const agree = results.filter(r => r.agreement).length;
  const single = results.filter(r => r.escalation.kind === "single_shot").length;
  const orch = total - single;
  const reasonHist: Record<string, number> = {};
  for (const r of results) if (r.escalation.reason) reasonHist[r.escalation.reason] = (reasonHist[r.escalation.reason] ?? 0) + 1;

  const md = `# Phase 65 Regression Backfill Report

Generated: ${new Date().toISOString()}
Sample: last ${DAYS} days, limit ${LIMIT}, processed ${total}

## Top-1 Agreement (v1 single-label vs v2 ranked[0].intent)

- Total: ${total}
- Agreement: ${agree} (${total ? ((agree / total) * 100).toFixed(1) : "0.0"}%)
- Disagreement: ${total - agree}

## Escalation Distribution (CORD-04 success criterion: ~80% single_shot)

- single_shot: ${single} (${total ? ((single / total) * 100).toFixed(1) : "0.0"}%)
- orchestrator: ${orch} (${total ? ((orch / total) * 100).toFixed(1) : "0.0"}%)

### Escalation reason breakdown

${Object.entries(reasonHist).map(([k, v]) => `- ${k}: ${v}`).join("\n") || "(none)"}

## Disagreement detail (top 20)

${results.filter(r => !r.agreement).slice(0, 20).map(r =>
`- email=${r.email_id}: v1=${r.v1_intent} v2=${r.v2_top1_intent} (v2 ranked_count=${r.v2_ranked_count}, escalation=${r.escalation.kind}${r.escalation.reason ? "/" + r.escalation.reason : ""})`
).join("\n")}

## Acceptance gate

- [${agree / total >= 0.7 ? "x" : " "}] >=70% top-1 agreement (sanity)
- [${single / total >= 0.7 ? "x" : " "}] >=70% single_shot (loose lower bound for CORD-04 ~80% target)
`;

  const reportPath = resolve(__dirname, "../.planning/phases/65-stage-3-ranked-multi-intent-coordinator/65-regression-report.md");
  writeFileSync(reportPath, md, "utf8");
  console.log(`Wrote ${reportPath}`);
  console.log(`Top-1 agreement: ${total ? ((agree / total) * 100).toFixed(1) : "0.0"}% | single_shot: ${total ? ((single / total) * 100).toFixed(1) : "0.0"}%`);
}

main().catch(e => { console.error(e); process.exit(1); });
```

import { createAdminClient } from "@/lib/supabase/admin";

export interface CoordinatorRunSummary {
  escalation_decision: "single_shot" | "orchestrator";
  escalation_reason: string | null;
  partial_synthesis: boolean;
}

export async function loadCoordinatorRunsForReview(
  automationRunIds: string[],
): Promise<Map<string, CoordinatorRunSummary>> {
  const result = new Map<string, CoordinatorRunSummary>();
  if (automationRunIds.length === 0) return result;
  const admin = createAdminClient();
  const { data, error } = await admin
    .from("coordinator_runs")
    .select("automation_run_id, escalation_decision, escalation_reason, partial_synthesis")
    .in("automation_run_id", automationRunIds);
  if (error) throw error;
  for (const r of data ?? []) {
    if (!r.automation_run_id) continue;
    result.set(r.automation_run_id, {
      escalation_decision: r.escalation_decision,
      escalation_reason: r.escalation_reason,
      partial_synthesis: r.partial_synthesis,
    });
  }
  return result;
}

}
