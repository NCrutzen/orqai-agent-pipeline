```typescript
const TTL_MS = 60_000;
const CACHE = new Map<string, { value: T; expires: number }>();

export async function loadX(admin: SupabaseClient, key: string): Promise<T> {
  const now = Date.now();
  const hit = CACHE.get(key);
  if (hit && hit.expires > now) return hit.value;
  const { data, error } = await admin.from(...).select(...).eq(...);
  if (error) {
    if (hit) return hit.value; // last-known-good
    return EMPTY;
  }
  const value = (data as T) ?? EMPTY;
  CACHE.set(key, { value, expires: now + TTL_MS });
  return value;
}

export function __resetCacheForTests(): void { CACHE.clear(); }
```
