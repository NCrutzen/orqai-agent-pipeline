```typescript
// Flexible schema for get_analytics_overview workspace response
// Uses .passthrough() because exact shape needs verification from live API
export const OrqaiWorkspaceSchema = z.object({
  total_deployments: z.number().optional(),
  total_requests: z.number().optional(),
  total_cost: z.number().optional(),
  total_tokens: z.number().optional(),
  avg_latency_ms: z.number().optional(),
  error_count: z.number().optional(),
  error_rate: z.number().optional(),
}).passthrough();

// Schema for per-agent metrics from query_analytics with group_by: agent_name
export const OrqaiAgentMetricSchema = z.object({
  agent_name: z.string(),
  requests: z.number().optional().default(0),
  cost: z.number().optional().default(0),
  latency_ms: z.number().optional().default(0),
  errors: z.number().optional().default(0),
}).passthrough();

export const OrqaiAgentMetricsArraySchema = z.array(OrqaiAgentMetricSchema);

export type OrqaiWorkspaceData = z.infer<typeof OrqaiWorkspaceSchema>;
export type OrqaiAgentMetric = z.infer<typeof OrqaiAgentMetricSchema>;
```

**1. Create `web/lib/inngest/functions/orqai-collector.ts`:**

Build an Inngest cron function that runs every hour. The function has 3 separate `step.run()` calls for retry isolation:

**Step 1: "fetch-workspace-overview"**
- Call Orq.ai MCP `get_analytics_overview` tool via the MCP adapter endpoint at `${process.env.NEXT_PUBLIC_APP_URL || "http://localhost:3000"}/api/mcp/mcp`
- IMPORTANT: Since MCP tools are called via the MCP server and we cannot directly import them in Inngest functions, use a direct HTTP call to the Orq.ai analytics API instead. The Orq.ai API base URL is `https://api.orq.ai` with the `ORQ_API_KEY` env var for authentication.
- Alternatively, if MCP is not directly callable from Inngest, make HTTP requests to the Orq.ai REST API:
  - `GET https://api.orq.ai/v2/analytics/overview` with header `Authorization: Bearer ${process.env.ORQ_API_KEY}`
  - `GET https://api.orq.ai/v2/analytics/query?metrics=usage,cost,latency,errors,agents,model_performance&group_by=agent_name`
- Validate response with `OrqaiWorkspaceSchema.safeParse(data)`
- If validation fails, log the raw response shape for debugging but do NOT throw -- store raw data with a flag
- Return the workspace data

**Step 2: "fetch-per-agent-metrics"**
- Call Orq.ai analytics API for per-agent breakdown
- Use `query_analytics` equivalent: request metrics grouped by `agent_name`
- Validate with `OrqaiAgentMetricsArraySchema.safeParse(data)`
- Return per-agent array

**Step 3: "store-snapshot"**
- Create admin Supabase client via `createAdminClient()`
- Insert into `orqai_snapshots` table:
  ```typescript
  {
  ```

Full function structure:
```typescript
import { inngest } from "@/lib/inngest/client";
import { createAdminClient } from "@/lib/supabase/admin";
import { OrqaiWorkspaceSchema, OrqaiAgentMetricsArraySchema } from "@/lib/orqai/types";

export const collectOrqaiAnalytics = inngest.createFunction(
  {
    id: "analytics/orqai-collect",
    retries: 3,
  },
  { cron: "0 * * * *" },  // Every hour
  async ({ step }) => {
    // Step 1: Fetch workspace overview
    const workspaceResult = await step.run("fetch-workspace-overview", async () => {
      const apiKey = process.env.ORQ_API_KEY;
      if (!apiKey) throw new Error("ORQ_API_KEY not configured");

      const res = await fetch("https://api.orq.ai/v2/analytics/overview", {
        headers: { Authorization: `Bearer ${apiKey}`, "Content-Type": "application/json" },
        signal: AbortSignal.timeout(30000),
      });

      if (!res.ok) throw new Error(`Orq.ai API error: ${res.status} ${res.statusText}`);

      const raw = await res.json();
      const parsed = OrqaiWorkspaceSchema.safeParse(raw);

      return {
        metrics: parsed.success ? parsed.data : null,
        raw: raw,
        validationOk: parsed.success,
        validationError: parsed.success ? null : parsed.error.message,
      };
    });

    // Step 2: Fetch per-agent metrics
    const agentResult = await step.run("fetch-per-agent-metrics", async () => {
      const apiKey = process.env.ORQ_API_KEY;
      if (!apiKey) throw new Error("ORQ_API_KEY not configured");

      const res = await fetch("https://api.orq.ai/v2/analytics/query?group_by=agent_name", {
        headers: { Authorization: `Bearer ${apiKey}`, "Content-Type": "application/json" },
        signal: AbortSignal.timeout(30000),
      });

      if (!res.ok) throw new Error(`Orq.ai API error: ${res.status} ${res.statusText}`);

      const raw = await res.json();
      // Response may be an array directly or wrapped in { data: [...] }
      const metricsArray = Array.isArray(raw) ? raw : (raw.data ?? raw.results ?? []);
      const parsed = OrqaiAgentMetricsArraySchema.safeParse(metricsArray);

      return {
        metrics: parsed.success ? parsed.data : null,
        raw: raw,
        validationOk: parsed.success,
      };
    });

    // Step 3: Store snapshot
    const snapshotId = await step.run("store-snapshot", async () => {
      const admin = createAdminClient();
      const ws = workspaceResult.metrics;

      const { data, error } = await admin.from("orqai_snapshots").insert({
        total_deployments: ws?.total_deployments ?? null,
        total_requests: ws?.total_requests ?? null,
        total_cost_usd: ws?.total_cost ?? null,
        total_tokens: ws?.total_tokens ?? null,
        avg_latency_ms: ws?.avg_latency_ms ?? null,
        error_count: ws?.error_count ?? null,
        error_rate_pct: ws?.error_rate ?? null,
        per_agent_metrics: agentResult.metrics ?? agentResult.raw,
        raw_workspace_data: workspaceResult.raw,
        raw_query_data: agentResult.raw,
        collected_at: new Date().toISOString(),
      }).select("id").single();

      if (error) throw new Error(`Failed to store snapshot: ${error.message}`);
      return data.id;
    });

    return { snapshotId, workspaceValid: workspaceResult.validationOk, agentValid: agentResult.validationOk };
  }
);
```

**IMPORTANT NOTES:**
- The Orq.ai analytics API endpoints (`/v2/analytics/overview` and `/v2/analytics/query`) need verification at runtime. If the exact paths differ, the first run will fail with a clear HTTP error and the paths can be adjusted.
- All side effects (fetch calls, DB writes) are inside `step.run()` per Inngest patterns.
- `retries: 3` handles transient API failures.
- 30-second timeout per API call (Orq.ai internal retry is 31s).
