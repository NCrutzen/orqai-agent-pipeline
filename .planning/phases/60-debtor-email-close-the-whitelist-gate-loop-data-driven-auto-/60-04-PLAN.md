    1. **`actions.ts`** — Server actions:
    ```typescript
    "use server";
    import { createAdminClient } from "@/lib/supabase/admin";
    import { revalidatePath } from "next/cache";

    export async function blockRule(ruleId: string) {
      const admin = createAdminClient();
      const { error } = await admin
        .from("classifier_rules")
        .update({ status: "manual_block", last_evaluated: new Date().toISOString() })
        .eq("id", ruleId);
      if (error) throw new Error(error.message);
      revalidatePath("/automations/classifier-rules");
    }

    export async function unblockRule(ruleId: string) {
      const admin = createAdminClient();
      // Returns rule to 'candidate' so cron re-evaluates; cron will promote/demote per gates
      const { error } = await admin
        .from("classifier_rules")
        .update({ status: "candidate", last_evaluated: new Date().toISOString() })
        .eq("id", ruleId);
      if (error) throw new Error(error.message);
      revalidatePath("/automations/classifier-rules");
    }
    ```

    2. **`page.tsx`** — Server component:
    ```typescript
    import { createAdminClient } from "@/lib/supabase/admin";
    import { RulesTable } from "./rules-table";

    export const dynamic = "force-dynamic";
    export const revalidate = 300; // 5-min interval per UI-SPEC

    export default async function ClassifierRulesPage() {
      const admin = createAdminClient();
      const mutate = process.env.CLASSIFIER_CRON_MUTATE === "true";

      const { data: rules } = await admin
        .from("classifier_rules")
        .select("*")
        .order("last_evaluated", { ascending: false });

      // Pull last 14 evaluations per rule for sparkline.
      const fourteenDaysAgo = new Date(Date.now() - 14 * 86400_000).toISOString();
      const { data: evals } = await admin
        .from("classifier_rule_evaluations")
        .select("swarm_type, rule_key, ci_lo, action, evaluated_at")
        .gte("evaluated_at", fourteenDaysAgo)
        .order("evaluated_at", { ascending: true });

      // Group evaluations by `${swarm_type}::${rule_key}` for the sparkline
      const evalsByRule = new Map<string, Array<{ ci_lo: number; evaluated_at: string }>>();
      for (const e of evals ?? []) {
        const k = `${e.swarm_type}::${e.rule_key}`;
        const arr = evalsByRule.get(k) ?? [];
        arr.push({ ci_lo: e.ci_lo, evaluated_at: e.evaluated_at });
        evalsByRule.set(k, arr);
      }

      return (
        <div className="px-8 pt-16 pb-12 max-w-[1280px] mx-auto">
          <h1 className="text-[28px] font-semibold leading-[1.2] font-[family-name:var(--font-cabinet)]">
            Classifier Rules
          </h1>
          <p className="text-[14px] leading-[1.5] text-[var(--v7-muted)] mt-2">
            Auto-promotion of regex and intent rules across all swarms. Wilson 95% CI-lo gates promotion at N≥30 and demotion at &lt;92%.
          </p>
          {!mutate && (
            <div className="mt-6 px-6 py-4 rounded-[var(--v7-radius-card)] bg-[var(--v7-blue-soft)] text-[var(--v7-blue)] text-[14px]">
              Shadow mode — cron records evaluations but does not mutate rule status. Showing &quot;would have promoted&quot; indicators.
            </div>
          )}
          <RulesTable
            rules={rules ?? []}
            evalsByRule={Object.fromEntries(evalsByRule)}
            shadowMode={!mutate}
          />
        </div>
      );
    }
    ```
