    ```typescript
    import { createAdminClient } from "@/lib/supabase/admin";
    import { AutomationRealtimeProvider } from "@/components/automations/automation-realtime-provider";
    import { QueueTree } from "./queue-tree";
    import { PredictedRowList } from "./predicted-row-list";

    export const dynamic = "force-dynamic";

    interface PageProps {
      searchParams: Promise<{
        topic?: string; entity?: string; mailbox?: string;
        rule?: string; tab?: string; before?: string;
      }>;
    }

    export default async function DebtorEmailReviewPage({ searchParams }: PageProps) {
      const params = await searchParams;
      const admin = createAdminClient();

      // D-13 counts via RPC (Supabase JS does not natively GROUP BY)
      const { data: counts } = await admin.rpc("classifier_queue_counts", {
        p_swarm_type: "debtor-email",
      });

      // D-14 cursor pagination
      let listQuery = admin
        .from("automation_runs")
        .select("*")
        .eq("status", "predicted")
        .eq("swarm_type", "debtor-email")
        .order("created_at", { ascending: false })
        .limit(100);
      if (params.before) listQuery = listQuery.lt("created_at", params.before);
      if (params.entity) listQuery = listQuery.eq("entity", params.entity);
      if (params.mailbox) listQuery = listQuery.eq("mailbox_id", parseInt(params.mailbox, 10));
      if (params.topic) listQuery = listQuery.eq("topic", params.topic);
      if (params.rule) listQuery = listQuery.eq("result->predicted->>rule", params.rule);
      const { data: rows } = await listQuery;

      // D-21 race-cohort: today's promoted rules
      const today = new Date();
      today.setHours(0, 0, 0, 0);
      const { data: promotedToday } = await admin
        .from("classifier_rules")
        .select("rule_key, promoted_at")
        .eq("swarm_type", "debtor-email")
        .eq("status", "promoted")
        .gte("promoted_at", today.toISOString());

      return (
        <AutomationRealtimeProvider automations={["debtor-email-review"]}>
          <div className="px-8 pt-16 pb-12 max-w-[1280px] mx-auto">
            <h1 className="text-[28px] font-semibold leading-[1.2] font-[family-name:var(--font-cabinet)]">
              Bulk Review
            </h1>
            <p className="text-[14px] leading-[1.5] text-[var(--v7-muted)] mt-2 mb-6">
              Review predicted classifications. Approved rows trigger Outlook categorize+archive and iController delete in the background.
            </p>
            <div className="grid grid-cols-[320px_1fr] gap-6">
              <QueueTree counts={counts ?? []} selection={params} />
              <PredictedRowList
                rows={rows ?? []}
                promotedToday={promotedToday ?? []}
                selection={params}
              />
            </div>
          </div>
        </AutomationRealtimeProvider>
      );
    }
    ```
