```typescript
export async function POST(req: NextRequest): Promise<NextResponse> {
  const sb = await createClient();
  const { data: { user } } = await sb.auth.getUser();
  if (!user) {
    return NextResponse.json({ error: "unauthenticated" }, { status: 401 });
  }
  let body: unknown;
  try { body = await req.json(); } catch {
    return NextResponse.json({ error: "invalid json" }, { status: 400 });

  const parsed = FeedbackPayload.safeParse(body);

    return NextResponse.json({ error: parsed.error.message }, { status: 400 });
  }
```

**Mirror for GET:**
- Replace `req.json()` with `req.nextUrl.searchParams.get('email_id') / get('stage')`.
- Define a sibling `FeedbackReadQuery = z.object({ email_id: z.string().uuid(), stage: z.union([z.literal(0), z.literal(1), z.literal(2), z.literal(3)]) })` and call `.safeParse({ email_id, stage: Number(stage) })`.

- Bucket rows: first row whose `operator_id === user.id` → `own_latest`; remaining rows → `others`.

**Mirror for `loadFeedbackMap(admin, emailIds, stage, viewerId)`:**
