    ```typescript
    // Idempotent: lowercase, strip non-[a-z0-9_], collapse underscores, trim.
    export function normalizeLabel(raw: string): string {
    ```

    ```typescript
    // Phase 86 D-02 + D-03 — nightly refresh of intent_proposal_clusters.
    // Cron: TZ=Europe/Amsterdam 0 4 (asterisk asterisk asterisk) daily 04:00 Amsterdam, 7 days/week.
    // Manual trigger: inngest.send({ name: "intent-proposals.refresh" }) (debounced 5min).
    // Cluster algorithm: Levenshtein 0.85 over normalised labels (web/lib/automations/intent-proposals/cluster.ts).
    // Replay-safety: every UUID and now() minted inside step.run (CLAUDE.md Phase 65 lock).

    import { inngest } from "@/lib/inngest/client";
    import { createAdminClient } from "@/lib/supabase/admin";
    import { clusterProposals } from "@/lib/automations/intent-proposals/cluster";
    import type { ProposalRow } from "@/lib/automations/intent-proposals/types";

    const DEBOUNCE_MS = 5 * 60 * 1000;
    const WINDOW_DAYS = 30;
    const SAMPLE_LIMIT = 5;
    const VIEWS_RETENTION_DAYS = 90;

    export const intentProposalsRefresh = inngest.createFunction(
      { id: "intent-proposals-refresh", name: "Intent proposals — daily refresh", retries: 3 },
      [
        { cron: "TZ=Europe/Amsterdam 0 4 * * *" },
        { event: "intent-proposals.refresh" },
      ],
      async ({ event, step }) => {
        const triggeredByEvent = event?.name === "intent-proposals.refresh";

        // Debounce only the event path; cron always runs.
        if (triggeredByEvent) {
          const skip = await step.run("debounce-check", async () => {
            const admin = createAdminClient();
            const { data } = await admin
              .from("intent_proposal_clusters")
              .select("refreshed_at")
              .order("refreshed_at", { ascending: false })
              .limit(1)
              .maybeSingle();
            if (!data?.refreshed_at) return false;
            const ageMs = Date.now() - new Date(data.refreshed_at).getTime();
            return ageMs &lt; DEBOUNCE_MS;
          });
          if (skip) return { skipped: "debounced" };
        }

        const proposals = await step.run("read-proposals", async () => {
          const admin = createAdminClient();
          const since = new Date(Date.now() - WINDOW_DAYS * 24 * 60 * 60 * 1000).toISOString();
          const { data, error } = await admin
            .from("intent_proposals_v1")
            .select("*")
            .gte("created_at", since);
          if (error) throw error;
          return (data ?? []) as ProposalRow[];
        });

        const upserted = await step.run("cluster-and-upsert", async () => {
          const admin = createAdminClient();
          const now = new Date();
          const windowEnd = now.toISOString();
          const windowStart = new Date(now.getTime() - WINDOW_DAYS * 24 * 60 * 60 * 1000).toISOString();

          // Group by swarm_type.
          const bySwarm = new Map&lt;string, ProposalRow[]&gt;();
          for (const row of proposals) {
            const arr = bySwarm.get(row.swarm_type) ?? [];
            arr.push(row);
            bySwarm.set(row.swarm_type, arr);
          }

          const rowsToUpsert: Array&lt;Record&lt;string, unknown&gt;&gt; = [];
          for (const [swarm, rows] of bySwarm) {
            const clusters = clusterProposals(rows, 0.85);
            for (const c of clusters) {
              rowsToUpsert.push({
                swarm_type: swarm,
                centroid_label: c.centroid,
                member_count: c.count,
                member_labels: c.memberLabels,
                sample_email_ids: c.members.slice(0, SAMPLE_LIMIT).map((m) =&gt; m.coordinator_run_id),
                window_start: windowStart,
                window_end: windowEnd,
                refreshed_at: windowEnd,
              });
            }
          }
          if (rowsToUpsert.length === 0) return 0;
          const { error } = await admin
            .from("intent_proposal_clusters")
            .upsert(rowsToUpsert, { onConflict: "swarm_type,centroid_label,window_end" });
          if (error) throw error;
          return rowsToUpsert.length;
        });

        const purged = await step.run("purge-old-views", async () =&gt; {
          const admin = createAdminClient();
          const cutoff = new Date(Date.now() - VIEWS_RETENTION_DAYS * 24 * 60 * 60 * 1000).toISOString();
          const { error, count } = await admin
            .from("intent_proposal_views")
            .delete({ count: "exact" })
            .lt("viewed_at", cutoff);
          if (error) throw error;
          return count ?? 0;
        });

        return { proposals: proposals.length, clusters_upserted: upserted, views_purged: purged };
      },
    );
    ```
