    ```typescript
    // Phase 86 D-04 + drift #3 — peer discovery tab strip.
    // Rendered BELOW StageTabStrip, NOT inside derive-stage-tabs.ts (that
    // file is RFC-architecture-locked to numeric stages 0..4). Extensible:
    // V9.0 Learning Inbox + V11.0 Handler queue land here as additional
    // DiscoveryTab entries when those phases ship.

    import Link from "next/link";
    import type { SwarmRow } from "@/lib/swarms/types";

    export interface DiscoveryTab {
      key: "intent-proposals";  // widen union as V9.0 / V11.0 tabs land
      label: string;
      slug: string;
      present: boolean;
    }

    export function deriveDiscoveryTabs(swarm: SwarmRow): DiscoveryTab[] {
      return [
        {
          key: "intent-proposals",
          label: "Intent proposals",
          slug: "intent-proposals",
          present: Boolean(swarm.stage3_coordinator_agent_key),
        },
      ];
    }

    interface Props {
      swarm: SwarmRow;
      current?: DiscoveryTab["key"];
    }

    export function DiscoveryTabStrip({ swarm, current }: Props) {
      const tabs = deriveDiscoveryTabs(swarm).filter((t) =&gt; t.present);
      if (tabs.length === 0) return null;
      return (
        &lt;nav
          className="discovery-tab-strip"
          style={{
            display: "flex",
            gap: "var(--space-2)",
            padding: "var(--space-2) var(--space-5)",
            background: "var(--v7-bg-2)",
            borderBottom: "1px solid var(--v7-border)",
          }}
        &gt;
          {tabs.map((t) =&gt; {
            const active = t.key === current;
            return (
              &lt;Link
                key={t.key}
                href={`/automations/${swarm.swarm_type}/${t.slug}`}
                aria-current={active ? "page" : undefined}
                style={{
                  fontSize: "13px",
                  fontWeight: active ? 500 : 400,
                  padding: "var(--space-2) var(--space-3)",
                  color: active ? "var(--v7-text)" : "var(--v7-text-muted)",
                  borderBottom: active
                    ? "2px solid var(--v7-brand-primary)"
                    : "2px solid transparent",
                  textDecoration: "none",
                }}
              &gt;
                {t.label}
              &lt;/Link&gt;
            );
          })}
        &lt;/nav&gt;
      );
    }
    ```

    ```typescript
    import { notFound } from "next/navigation";
    import { createAdminClient } from "@/lib/supabase/admin";
    import { loadSwarm } from "@/lib/swarms/registry";
    import { PageHeader } from "../_shell/page-header";
    import { StageTabStrip } from "../_shell/stage-tab-strip";
    import { DiscoveryTabStrip } from "../_shell/discovery-tab-strip";
    import { IntentProposalsClientShell } from "./client-shell";
    import type { ClusterRow } from "@/lib/automations/intent-proposals/types";

    export const dynamic = "force-dynamic";

    interface PageProps {
      params: Promise&lt;{ swarm: string }&gt;;
      searchParams: Promise&lt;{ swarm_filter?: "current" | "all" }&gt;;
    }

    export default async function IntentProposalsPage({ params, searchParams }: PageProps) {
      const { swarm: swarmSlug } = await params;
      const { swarm_filter = "current" } = await searchParams;
      const swarm = await loadSwarm(swarmSlug);
      if (!swarm || !swarm.stage3_coordinator_agent_key) return notFound();

      const admin = createAdminClient();
      let query = admin
        .from("intent_proposal_clusters")
        .select("*")
        .order("member_count", { ascending: false })
        .order("refreshed_at", { ascending: false })
        .limit(50);
      if (swarm_filter === "current") query = query.eq("swarm_type", swarm.swarm_type);
      const { data, error } = await query;
      if (error) throw error;
      const clusters = (data ?? []) as ClusterRow[];

      return (
        &lt;&gt;
          &lt;PageHeader swarm={swarm} /&gt;
          &lt;StageTabStrip swarm={swarm} currentStage={0 /* none active */ as 0} /&gt;
          &lt;DiscoveryTabStrip swarm={swarm} current="intent-proposals" /&gt;
          &lt;IntentProposalsClientShell
            swarmType={swarm.swarm_type}
            filter={swarm_filter}
            clusters={clusters}
          /&gt;
        &lt;/&gt;
      );
    }
    ```
    (Note: if the StageTabStrip's `currentStage` is required and rejects "no active", pass a sentinel like `-1 as never` adapted to its prop type, or accept that no stage tab is highlighted while the discovery tab is active. Adapt to the actual StageTabStrip prop signature at execution time.)

    ```typescript
    "use server";
    import { createAdminClient } from "@/lib/supabase/admin";
    import { inngest } from "@/lib/inngest/client";

    export interface LogTabViewInput {
      swarm_type: string | null;
      cluster_id?: string | null;
      operator_id?: string | null;
      user_agent?: string | null;
    }

    export async function logTabView(input: LogTabViewInput): Promise&lt;void&gt; {
      const admin = createAdminClient();
      const { error } = await admin.from("intent_proposal_views").insert({
        swarm_type: input.swarm_type,
        cluster_id: input.cluster_id ?? null,
        operator_id: input.operator_id ?? null,
        user_agent: input.user_agent ?? null,
      });
      if (error) throw error;
    }

    export async function triggerRefresh(): Promise&lt;{ ok: true }&gt; {
      // NOTE: do NOT destructure inngest.send — Phase 65 this-binding pitfall.
      await inngest.send({ name: "intent-proposals.refresh" });
      return { ok: true };
    }
    ```
