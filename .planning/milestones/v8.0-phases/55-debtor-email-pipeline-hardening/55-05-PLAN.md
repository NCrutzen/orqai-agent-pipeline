```typescript
/**
 * Prompt version anchors for intent-agent + body-agent.
 * Short git-sha at build time — see Phase 55 CONTEXT §agent_runs schema & versioning.
 *
 * Priority:
 *  1. VERCEL_GIT_COMMIT_SHA (injected on Vercel builds) sliced to 7 chars
 *  2. `git rev-parse --short HEAD` read at module-load time (dev server / local scripts)
 *  3. Literal "dev" (tests / CI without git)
 */
function resolveShortSha(): string {
  const fromVercel = process.env.VERCEL_GIT_COMMIT_SHA;
  if (fromVercel && fromVercel.length >= 7) return fromVercel.slice(0, 7);

  // Attempt local git only in server-side Node contexts. Swallow any error:
  // edge/browser/test environments should land on "dev".
  try {
    if (typeof process !== "undefined" && process.versions?.node) {
      // Using dynamic require to avoid bundler including child_process on edge.
      // eslint-disable-next-line @typescript-eslint/no-require-imports
      const cp = require("node:child_process") as typeof import("node:child_process");
      const sha = cp.execSync("git rev-parse --short=7 HEAD", { encoding: "utf8", stdio: ["ignore", "pipe", "ignore"] }).trim();
      if (/^[a-f0-9]{7}$/.test(sha)) return sha;
    }
  } catch {
    /* fall through */
  }
  return "dev";
}

const SHORT_SHA = resolveShortSha();

export const INTENT_VERSION: string = SHORT_SHA;
export const BODY_VERSION: string = SHORT_SHA;
```

```typescript
import type { SupabaseClient } from "@supabase/supabase-js";
import type { Entity, Status } from "./types";

type JsonValue =
  | string
  | number
  | boolean
  | null
  | JsonValue[]
  | { [k: string]: JsonValue };

const SWARM_TYPE = "debtor_email" as const;

export type CreateRunInput = {
  email_id: string;
  inngest_run_id: string;
  entity: Entity;
  status?: Status;
  automation_run_id?: string | null;
  context_extra?: Record<string, JsonValue>;
};

export async function createRun(
  supabase: SupabaseClient,
  input: CreateRunInput,
): Promise<string> {
  const { data, error } = await supabase
    .from("agent_runs")           // public schema (default)
    .insert({
      swarm_type: SWARM_TYPE,
      email_id: input.email_id,
      inngest_run_id: input.inngest_run_id,
      automation_run_id: input.automation_run_id ?? null,
      status: input.status ?? "classifying",
      context: { entity: input.entity, ...(input.context_extra ?? {}) },
    })
    .select("id")
    .single();

  if (error) throw new Error(`createRun: ${error.message}`);
  return (data as { id: string }).id;
}

export type UpdateRunPatch = Partial<{
  status: Status;
  intent: string;
  sub_type: string | null;
  document_reference: string | null;
  language: string;
  confidence: string;
  urgency: string;
  intent_version: string;
  reasoning: string;
  body_version: string;
  detected_tone: string;
  draft_url: string;
  completed_at: string;
  // tool_outputs intentionally NOT here — use mergeToolOutputs.
  // context updates → use mergeContext below.
}>;

export async function updateRun(
  supabase: SupabaseClient,
  id: string,
  patch: UpdateRunPatch,
): Promise<void> {
  if (Object.keys(patch).length === 0) return;
  const { error } = await supabase
    .from("agent_runs")
    .update(patch)
    .eq("id", id);
  if (error) throw new Error(`updateRun: ${error.message}`);
}

export async function mergeContext(
  supabase: SupabaseClient,
  id: string,
  patch: Record<string, JsonValue>,
): Promise<void> {
  const { data, error: readErr } = await supabase
    .from("agent_runs")
    .select("context")
    .eq("id", id)
    .single();
  if (readErr) throw new Error(`mergeContext read: ${readErr.message}`);
  const current = ((data as { context: Record<string, JsonValue> | null } | null)?.context as Record<string, JsonValue> | null) ?? {};
  const merged = { ...current, ...patch };
  const { error: writeErr } = await supabase
    .from("agent_runs")
    .update({ context: merged })
    .eq("id", id);
  if (writeErr) throw new Error(`mergeContext write: ${writeErr.message}`);
}

export async function mergeToolOutputs(
  supabase: SupabaseClient,
  id: string,
  stage: string,
  payload: JsonValue,
): Promise<void> {
  const { data, error: readErr } = await supabase
    .from("agent_runs")
    .select("tool_outputs")
    .eq("id", id)
    .single();
  if (readErr) throw new Error(`mergeToolOutputs read: ${readErr.message}`);

  const current =
    ((data as { tool_outputs: Record<string, JsonValue> | null } | null)?.tool_outputs as
      | Record<string, JsonValue>
      | null) ?? {};
  const merged: Record<string, JsonValue> = { ...current, [stage]: payload };

  const { error: writeErr } = await supabase
    .from("agent_runs")
    .update({ tool_outputs: merged })
    .eq("id", id);
  if (writeErr) throw new Error(`mergeToolOutputs write: ${writeErr.message}`);
}

export async function findCachedOutput<T = Record<string, unknown>>(
  supabase: SupabaseClient,
  email_id: string,
  version_field: "intent_version" | "body_version",
  version_value: string,
  output_field: "tool_outputs" | "draft_url",
): Promise<T | null> {
  const { data, error } = await supabase
    .from("agent_runs")
    .select(`id,${output_field}`)
    .eq("email_id", email_id)
    .eq("swarm_type", SWARM_TYPE)
    .eq(version_field, version_value)
    .not(output_field, "is", null)
    .order("created_at", { ascending: false })
    .limit(1)
    .maybeSingle();

  if (error) throw new Error(`findCachedOutput: ${error.message}`);
  if (!data) return null;
  return (data as Record<string, unknown>)[output_field] as T | null;
}
```
