<objective>
Make `createIcontrollerDraft` safe to call multiple times for the same (graph_message_id, entity) — a blocker for the copy-document swarm launch. Three pieces:
1. **Sentinel-row idempotency** via `debtor.icontroller_drafts` (Plan 01).
2. **Operator marker** = HTML comment injected into the draft body via `page.evaluate()` (shadow-DOM safe).
3. **Orphan reconcile** — cleanup-worker periodically deletes stale sentinels (claimed_at > 15min, draft_url NULL) so retries can re-create drafts.

Purpose: Without idempotency, retry storms duplicate drafts in iController. Without the marker, Phase 2 body-diff scan can't correlate edited drafts back to `public.agent_runs`. Without reconcile, a single Browserless crash creates a permanent ghost row.

Output: One new repository module, 2 edited files, 1 cleanup-worker addition, 3 test files flipped to real assertions.
</objective>

```typescript
import type { SupabaseClient } from "@supabase/supabase-js";

export type ClaimResult =
  | { status: "claimed"; id: string }
  | { status: "already_drafted"; existing_id: string; existing_draft_url: string | null };

export interface ClaimInput {
  graph_message_id: string;
  entity: string;
  agent_run_id: string | null;
}

/**
 * Atomically claim a sentinel row for (graph_message_id, entity). Relies on
 * UNIQUE(graph_message_id, entity) in debtor.icontroller_drafts (Phase 55 migration).
 * INSERT ON CONFLICT DO NOTHING + a follow-up SELECT gives us the existing id on conflict.
 */
export async function claimDraftSentinel(
  admin: SupabaseClient,
  input: ClaimInput,
): Promise<ClaimResult> {
  const { data: inserted, error: insErr } = await admin
    .schema("debtor")
    .from("icontroller_drafts")
    .insert({
      graph_message_id: input.graph_message_id,
      entity: input.entity,
      agent_run_id: input.agent_run_id,
    })
    .select("id")
    .maybeSingle();

  if (insErr && insErr.code !== "23505") {
    throw new Error(`claimDraftSentinel insert: ${insErr.message}`);
  }
  if (inserted?.id) {
    return { status: "claimed", id: inserted.id };
  }

  // Conflict hit — fetch existing row.
  const { data: existing, error: selErr } = await admin
    .schema("debtor")
    .from("icontroller_drafts")
    .select("id, icontroller_draft_url")
    .eq("graph_message_id", input.graph_message_id)
    .eq("entity", input.entity)
    .maybeSingle();
  if (selErr || !existing) {
    throw new Error(`claimDraftSentinel select-after-conflict: ${selErr?.message ?? "no row"}`);
  }
  return {
    status: "already_drafted",
    existing_id: existing.id,
    existing_draft_url: existing.icontroller_draft_url,
  };
}

export async function markDraftSaved(
  admin: SupabaseClient,
  input: { id: string; icontroller_draft_url: string; icontroller_message_id?: string | null },
): Promise<void> {
  const { error } = await admin
    .schema("debtor")
    .from("icontroller_drafts")
    .update({
      icontroller_draft_url: input.icontroller_draft_url,
      icontroller_message_id: input.icontroller_message_id ?? null,
    })
    .eq("id", input.id);
  if (error) throw new Error(`markDraftSaved: ${error.message}`);
}

export async function releaseDraftSentinel(
  admin: SupabaseClient,
  id: string,
): Promise<void> {
  const { error } = await admin
    .schema("debtor")
    .from("icontroller_drafts")
    .delete()
    .eq("id", id);
  if (error) throw new Error(`releaseDraftSentinel: ${error.message}`);
}

/**
 * Orphan = sentinel claimed >15min ago, Browserless crashed before it could
 * save → no draft URL, no deletion. Cleanup-worker calls this + DELETEs
 * the orphans so the next create-draft retry can re-claim.
 */
export async function findOrphanSentinels(
  admin: SupabaseClient,
  olderThanMs = 15 * 60 * 1000,
): Promise<Array<{ id: string; graph_message_id: string; entity: string; claimed_at: string }>> {
  const cutoff = new Date(Date.now() - olderThanMs).toISOString();
  const { data, error } = await admin
    .schema("debtor")
    .from("icontroller_drafts")
    .select("id, graph_message_id, entity, claimed_at")
    .is("icontroller_draft_url", null)
    .is("deleted_at", null)
    .lt("claimed_at", cutoff)
    .limit(500);
  if (error) throw new Error(`findOrphanSentinels: ${error.message}`);
  return data ?? [];
}

export function buildOperatorMarker(agentRunId: string): string {
  return `<!-- MR-AUTOMATION-DRAFT v1 run=${agentRunId} do-not-edit-above -->`;
}
```
