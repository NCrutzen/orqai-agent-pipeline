```typescript
deriveTags: (runs) => {
  const out: string[] = [];

  // Existing behaviour — keep category tag first.
  const cat = runs
    .map((r) => extractDebtorEmailCategory(r))
    .find((c): c is string => !!c);
  if (cat) out.push(cat);

  // New Phase 55: provenance chips.
  const reasons = new Set<string>();
  for (const run of runs) {
    const r = (run.result ?? {}) as Record<string, unknown>;

    // Explicit (persisted by ingest route / intent-agent).
    const explicit = Array.isArray((r as { review_reason?: unknown }).review_reason)
      ? ((r as { review_reason: unknown[] }).review_reason.filter((x) => typeof x === "string") as string[])
      : [];
    for (const tag of explicit) reasons.add(tag);

    // regex:unknown / regex:low-conf
    const predicted = r.predicted as { rule?: string; confidence?: number } | undefined;
    if (predicted?.rule === "no_match") reasons.add("regex:unknown");
    else if (typeof predicted?.confidence === "number" && predicted.confidence < 0.8) {
      reasons.add("regex:low-conf");
    }

    // llm:* from intent-agent output (written into result by triage-fire step)
    const intent = r.intent_agent as { confidence?: number; intent?: string } | undefined;
    if (typeof intent?.confidence === "number" && intent.confidence < 0.8) {
      reasons.add("llm:low-conf");
    }
    if (intent?.intent === "other") reasons.add("llm:no-intent");

    // ingest:skipped-leak — any run reaching us with this status
    if (run.status === "skipped_not_whitelisted") reasons.add("ingest:skipped-leak");

    // draft:pending — icontroller_delete pending or drafter deferred
    if (
      r.stage === "icontroller_delete" &&
      (r as { icontroller?: string }).icontroller === "pending"
    ) {
      reasons.add("draft:pending");
    }
    if (run.automation === "debtor-email-drafter" && run.status === "deferred") {
      reasons.add("draft:pending");
    }
  }
  for (const r of reasons) out.push(r);

  return out;
},
```

```typescript
"use client";

import { useState, useTransition } from "react";
import Link from "next/link";
import { ExternalLink, ThumbsDown, ThumbsUp } from "lucide-react";
import { toast } from "sonner";

type AgentRun = {
  id: string;
  swarm_type: string;
  intent: string | null;
  sub_type: string | null;
  document_reference: string | null;
  confidence: string | null;
  draft_url: string | null;
  tool_outputs: Record<string, unknown> | null;
  context: Record<string, unknown> | null;
  human_verdict: string | null;
  created_at: string;
};

const REJECT_REASONS = [
  { value: "rejected_wrong_intent", label: "Wrong intent" },
  { value: "rejected_wrong_attachment", label: "Wrong attachment" },
  { value: "rejected_wrong_language", label: "Wrong language" },
  { value: "rejected_wrong_tone", label: "Wrong tone" },
  { value: "rejected_other", label: "Other (use notes)" },
];

export function VerdictShell({ run }: { run: AgentRun }) {
  const [verdict, setVerdict] = useState<"approved" | "rejected" | null>(null);
  const [reason, setReason] = useState<string>("");
  const [notes, setNotes] = useState<string>("");
  const [isPending, startTransition] = useTransition();

  const ctx = (run.context ?? {}) as Record<string, unknown>;
  const iControllerDraftUrl =
    typeof ctx.icontroller_draft_url === "string" ? ctx.icontroller_draft_url : run.draft_url;
  const emailSubject = typeof ctx.email_subject === "string" ? ctx.email_subject : "(no subject)";
  const emailFrom = typeof ctx.email_from === "string" ? ctx.email_from : "(unknown sender)";

  const tools = (run.tool_outputs ?? {}) as Record<string, unknown>;
  const generated = tools.body as { bodyHtml?: string } | undefined;
  const preview = (generated?.bodyHtml ?? "").replace(/<[^>]+>/g, " ").slice(0, 300);

  const submit = () => {
    if (verdict === "rejected" && !reason) {
      toast.error("Select a reason for rejection");
      return;
    }
    startTransition(async () => {
      const res = await fetch("/api/automations/debtor/verdict", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          run_id: run.id,
          verdict: verdict === "approved" ? "approved" : reason,
          reason: verdict === "rejected" ? reason : null,
          notes: notes.trim() || null,
        }),
      });
      if (!res.ok) {
        const body = await res.json().catch(() => ({}));
        toast.error(body.error ?? "Failed to save verdict");
        return;
      }
      toast.success("Verdict saved");
      setVerdict(null);
      setReason("");
      setNotes("");
    });
  };

  if (run.human_verdict) {
    return (
      <div className="p-6 max-w-3xl">
        <h1 className="font-[var(--font-cabinet)] text-xl mb-2">Already reviewed</h1>
        <p className="text-[var(--v7-muted)]">This run was labeled {run.human_verdict} and is no longer open for review.</p>
      </div>
    );
  }

  return (
    <div className="p-6 max-w-3xl space-y-4">
      <header>
        <h1 className="font-[var(--font-cabinet)] text-xl mb-1">{emailSubject}</h1>
        <p className="text-sm text-[var(--v7-muted)]">From: {emailFrom}</p>
        <p className="text-xs text-[var(--v7-faint)]">
          Intent: {run.intent ?? "—"} · Sub-type: {run.sub_type ?? "—"} · Ref: {run.document_reference ?? "—"} · Confidence: {run.confidence ?? "—"}
        </p>
      </header>

      <section className="border border-[var(--v7-line)] rounded-[var(--v7-radius-mini)] p-4">
        <h2 className="text-sm font-medium mb-2">Draft preview</h2>
        <p className="text-sm whitespace-pre-wrap text-[var(--v7-text)]">{preview || "(no body generated)"}</p>
        {iControllerDraftUrl && (
          <Link
            href={iControllerDraftUrl}
            target="_blank"
            className="inline-flex items-center gap-1 mt-3 text-xs text-[var(--v7-teal)] hover:underline"
          >
            <ExternalLink size={12} />
            Open in iController
          </Link>
        )}
      </section>

      <section className="flex items-center gap-3">
        <button
          type="button"
          onClick={() => setVerdict("approved")}
          className={`inline-flex items-center gap-2 px-4 py-2 rounded-[var(--v7-radius-mini)] border ${
            verdict === "approved" ? "bg-[var(--v7-teal-soft)] border-[var(--v7-teal)]" : "border-[var(--v7-line)]"
          }`}
        >
          <ThumbsUp size={16} /> Approve
        </button>
        <button
          type="button"
          onClick={() => setVerdict("rejected")}
          className={`inline-flex items-center gap-2 px-4 py-2 rounded-[var(--v7-radius-mini)] border ${
            verdict === "rejected" ? "bg-red-50 border-red-500" : "border-[var(--v7-line)]"
          }`}
        >
          <ThumbsDown size={16} /> Reject
        </button>
      </section>

      {verdict === "rejected" && (
        <section className="space-y-2">
          <label className="block text-sm">Reason (required)</label>
          <select
            value={reason}
            onChange={(e) => setReason(e.target.value)}
            className="w-full border border-[var(--v7-line)] rounded-[var(--v7-radius-mini)] px-3 py-2 text-sm"
          >
            <option value="">Select a reason…</option>
            {REJECT_REASONS.map((r) => (
              <option key={r.value} value={r.value}>{r.label}</option>
            ))}
          </select>
        </section>
      )}

      {verdict !== null && (
        <section className="space-y-2">
          <label className="block text-sm">Notes (optional)</label>
          <textarea
            value={notes}
            onChange={(e) => setNotes(e.target.value)}
            className="w-full border border-[var(--v7-line)] rounded-[var(--v7-radius-mini)] px-3 py-2 text-sm"
            rows={3}
          />
        </section>

      {verdict !== null && (
        <button
          type="button"
          onClick={submit}
          disabled={isPending || (verdict === "rejected" && !reason)}
          className="px-4 py-2 rounded-[var(--v7-radius-mini)] bg-[var(--v7-teal)] text-[var(--v7-inverse)] disabled:opacity-50"
        >
          {isPending ? "Saving…" : "Submit verdict"}
        </button>
      )}
    </div>
  );
}
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import { createServerClient } from "@/lib/supabase/server";
import { createAdminClient } from "@/lib/supabase/admin";

const HUMAN_VERDICTS = [
  "approved",
  "edited_minor",
  "edited_major",
  "rejected_wrong_intent",
  "rejected_wrong_reference",
  "rejected_wrong_attachment",
  "rejected_wrong_language",
  "rejected_wrong_tone",
  "rejected_other",
] as const;

const REJECT_REASONS = [
  "rejected_wrong_intent",
  "rejected_wrong_attachment",
  "rejected_wrong_language",
  "rejected_wrong_tone",
  "rejected_other",
] as const;

const VerdictSchema = z
  .object({
    run_id: z.string().uuid(),
    verdict: z.enum(HUMAN_VERDICTS),
    reason: z.enum(REJECT_REASONS).nullable().optional(),
    notes: z.string().max(2000).nullable().optional(),
  })
  .superRefine((val, ctx) => {
    // Rejection requires a reason.
    if (val.verdict !== "approved" && val.verdict !== "edited_minor" && val.verdict !== "edited_major") {
      if (!val.reason) {
        ctx.addIssue({
          code: z.ZodIssueCode.custom,
          path: ["reason"],
          message: "reason is required when verdict is a rejection",
        });
      }
    }
  });

export async function POST(request: NextRequest) {
  // Auth: reuse dashboard session.
  const serverClient = await createServerClient();
  const { data: { user } } = await serverClient.auth.getUser();
  if (!user) {
    return NextResponse.json({ error: "unauthorized" }, { status: 401 });
  }

  const body = await request.json().catch(() => null);
  const parsed = VerdictSchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json(
      { error: "validation_failed", details: parsed.error.flatten() },
      { status: 400 },
    );
  }

  const { run_id, verdict, reason, notes } = parsed.data;
  const admin = createAdminClient();
  const { error } = await admin
    .from("agent_runs")
    .update({
      human_verdict: verdict,
      verdict_reason: reason ?? null,
      human_notes: notes ?? null,
      verdict_set_at: new Date().toISOString(),
      verdict_set_by: user.email ?? user.id,
    })
    .eq("id", run_id);

  if (error) {
    return NextResponse.json({ error: "update_failed", details: error.message }, { status: 500 });
  }

  return NextResponse.json({ ok: true });
}
```
