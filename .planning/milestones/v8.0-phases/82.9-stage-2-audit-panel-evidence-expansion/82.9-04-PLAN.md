    function renderInputs(inputs: Stage2InputsView | null | undefined): JSX.Element {
      if (inputs == null) {

      }
      switch (inputs.kind) {
        case "thread_inheritance":
          return (
            <>
              <InputRow label="Prior label" value={inputs.prior_email_label_id} />
              <InputRow label="Conversation" value={inputs.conversation_id} />
            </>
          );
        case "sender_match":
          return <InputRow label="Sender" value={inputs.sender_email} />;
        case "identifier_match":
          return (
            <InputRow
              label="Matched IDs"
              value={
                <div style={{ display: "flex", gap: "6px", flexWrap: "wrap" }}>
                  {inputs.matched_identifiers.map((id) => (
                    <span key={id} style={metaChipStyle}>{id}</span>
                  ))}
                </div>
              }
            />
          );
        case "llm_tiebreaker":
          return (
            <>
              {inputs.sender_email && <InputRow label="Sender" value={inputs.sender_email} />}
              <InputRow
                label="Matched IDs"
                value={
                  <div style={{ display: "flex", gap: "6px", flexWrap: "wrap" }}>
                    {inputs.matched_identifiers.map((id) => (
                      <span key={id} style={metaChipStyle}>{id}</span>
                    ))}
                  </div>
                }
              />
            </>
          );
        case "unresolved":
          return (
            <>
              {inputs.sender_email && <InputRow label="Sender" value={inputs.sender_email} />}
              <InputRow
                label="Matched IDs"
                value={

                    ? <span style={mutedStyle}><em>none</em></span>
                    : (
                      <div style={{ display: "flex", gap: "6px", flexWrap: "wrap" }}>
                        {inputs.matched_identifiers.map((id) => (
                          <span key={id} style={metaChipStyle}>{id}</span>

                    )

              />
            </>
          );
      }
