    ```typescript
    // cron: TZ=Europe/Amsterdam 0 6 * * 1-5 — daily 06:00 Amsterdam, Mon-Fri (D-09)
    // Reads public.classifier_rule_telemetry, writes public.classifier_rule_evaluations.
    // Mutates public.classifier_rules.status only when CLASSIFIER_CRON_MUTATE === "true".
    import { inngest } from "@/lib/inngest/client";
    import { createAdminClient } from "@/lib/supabase/admin";
    import {
      wilsonCiLower,
      shouldPromote,
      shouldDemote,
    } from "@/lib/classifier/wilson";
    import type { EvaluationAction } from "@/lib/classifier/types";

    interface TelemetryRow { swarm_type: string; rule_key: string; n: number; agree: number; }
    interface RuleRow { swarm_type: string; rule_key: string; status: "candidate"|"promoted"|"demoted"|"manual_block"; }

    export const classifierPromotionCron = inngest.createFunction(
      { id: "classifier/promotion-cron", retries: 2 },
      { cron: "TZ=Europe/Amsterdam 0 6 * * 1-5" },
      async ({ step }) => {
        const mutate = process.env.CLASSIFIER_CRON_MUTATE === "true";
        const admin = createAdminClient();

        const telemetry = await step.run("load-telemetry", async () => {
          const { data, error } = await admin
            .from("classifier_rule_telemetry")
            .select("swarm_type, rule_key, n, agree");
          if (error) throw new Error(`telemetry load failed: ${error.message}`);
          return (data ?? []) as TelemetryRow[];
        });

        const rules = await step.run("load-rules", async () => {
          const { data, error } = await admin
            .from("classifier_rules")
            .select("swarm_type, rule_key, status");
          if (error) throw new Error(`rules load failed: ${error.message}`);
          return (data ?? []) as RuleRow[];
        });

        const ruleByKey = new Map<string, RuleRow>(
          rules.map((r) => [`${r.swarm_type}::${r.rule_key}`, r]),
        );

        const results: Array<{ swarm_type: string; rule_key: string; action: EvaluationAction; ci_lo: number }> = [];

        for (const t of telemetry) {
          const stepId = `eval-${t.swarm_type}-${t.rule_key}`;
          const result = await step.run(stepId, async () => {
            const ci_lo = wilsonCiLower(t.n, t.agree);
            const rule = ruleByKey.get(`${t.swarm_type}::${t.rule_key}`);
            const status = rule?.status ?? "candidate";

            // Manual blocks are never auto-touched
            if (status === "manual_block") {
              await admin
                .from("classifier_rule_evaluations")
                .upsert(
                  { swarm_type: t.swarm_type, rule_key: t.rule_key, n: t.n, agree: t.agree, ci_lo, action: "no_change" satisfies EvaluationAction },
                  { onConflict: "swarm_type,rule_key,evaluated_at" },
                );
              return { swarm_type: t.swarm_type, rule_key: t.rule_key, action: "no_change" as EvaluationAction, ci_lo };
            }

            const promote = status === "candidate" && shouldPromote(t.n, ci_lo);
            const demote  = status === "promoted"  && shouldDemote(ci_lo);

            let action: EvaluationAction = "no_change";
            if (mutate) {
              if (promote) {
                action = "promoted";
                await admin.from("classifier_rules").update({
                  status: "promoted", promoted_at: new Date().toISOString(),
                  n: t.n, agree: t.agree, ci_lo, last_evaluated: new Date().toISOString(),
                }).eq("swarm_type", t.swarm_type).eq("rule_key", t.rule_key);
              } else if (demote) {
                action = "demoted";
                console.warn("[classifier-cron] DEMOTION", { swarm_type: t.swarm_type, rule_key: t.rule_key, n: t.n, ci_lo });
                await admin.from("classifier_rules").update({
                  status: "demoted", last_demoted_at: new Date().toISOString(),
                  n: t.n, agree: t.agree, ci_lo, last_evaluated: new Date().toISOString(),
                }).eq("swarm_type", t.swarm_type).eq("rule_key", t.rule_key);
              } else {
                // counters refresh
                await admin.from("classifier_rules").update({
                  n: t.n, agree: t.agree, ci_lo, last_evaluated: new Date().toISOString(),
                }).eq("swarm_type", t.swarm_type).eq("rule_key", t.rule_key);
              }
            } else {
              // SHADOW MODE — never mutate classifier_rules
              if (promote) action = "shadow_would_promote";
              else if (demote) action = "shadow_would_demote";
              else action = "no_change";
            }

            await admin
              .from("classifier_rule_evaluations")
              .upsert(
                { swarm_type: t.swarm_type, rule_key: t.rule_key, n: t.n, agree: t.agree, ci_lo, action },
                { onConflict: "swarm_type,rule_key,evaluated_at" },
              );

            return { swarm_type: t.swarm_type, rule_key: t.rule_key, action, ci_lo };
          });
          results.push(result);
        }

        return { mutate, evaluated: results.length, results };
      },
    );
    ```
