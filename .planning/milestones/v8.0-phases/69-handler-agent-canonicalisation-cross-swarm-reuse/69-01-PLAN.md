```typescript
import type { SupabaseClient } from "@supabase/supabase-js";
import { loadSwarm } from "./registry";

export interface BrandRegister {
  code: string;
  display_name: string;
  register_language: "nl" | "fr" | "en" | "de";
  register_dialect: string;
  signoff_phrase: string;
  formal_address: string;
  nxt_database_alias: string;
  icontroller_company: string;
}

const ENTITY_BRAND_CACHE = new Map<
  string,
  { value: BrandRegister[]; expires: number }
>();
const TTL_MS = 60_000;

/**
 * Phase 69 D-11. Load all BrandRegister rows for a swarm_type.
 * Routes through loadSwarm (which is itself TTL-cached). Last-known-good
 * fallback on Supabase error. Throws on string-shape rows (migration
 * 20260505a not yet applied).
 */
export async function loadAllBrandRegisters(
  admin: SupabaseClient,
  swarmType: string,
): Promise<BrandRegister[]> {
  const now = Date.now();
  const hit = ENTITY_BRAND_CACHE.get(swarmType);
  if (hit && hit.expires > now) return hit.value;

  const swarm = await loadSwarm(admin, swarmType);
  const raw = swarm?.entity_brand;
  if (!Array.isArray(raw)) {
    if (hit) return hit.value;
    return [];
  }
  if (raw.length > 0 && typeof raw[0] !== "object") {
    throw new Error(
      `swarms.entity_brand for swarm_type='${swarmType}' is still string-shape; run migration 20260505a_entity_brand_expansion.sql`,
    );
  }
  const value = raw as BrandRegister[];
  ENTITY_BRAND_CACHE.set(swarmType, { value, expires: now + TTL_MS });
  return value;
}

/**
 * Phase 69 D-11 / D-12. Resolve a single brand_code to its BrandRegister.
 * Throws on unknown brand_code — no defensive fallback (Phase 68 D-12
 * precedent; misconfiguration must surface in Bulk Review).
 */
export async function loadBrandRegister(
  admin: SupabaseClient,
  swarmType: string,
  brandCode: string,
): Promise<BrandRegister> {
  const all = await loadAllBrandRegisters(admin, swarmType);
  const found = all.find((b) => b.code === brandCode);
  if (!found) {
    throw new Error(
      `brand_register lookup failed: swarm_type='${swarmType}' brand_code='${brandCode}' not in registry`,
    );
  }
  return found;
}

/** Test-only. Clears the module-level cache. */
export function __resetCacheForTests(): void {
  ENTITY_BRAND_CACHE.clear();
}
```

    }
