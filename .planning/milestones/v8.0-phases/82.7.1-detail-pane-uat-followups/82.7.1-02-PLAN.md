**Step 2: Add a useEffect that schedules deferred unmount 180ms after a row enters `pendingRemovalIds`.** 180ms = 150ms fade + 30ms slop:

```typescript
useEffect(() => {
  if (pendingRemovalIds.size === 0) return;
  const timers: number[] = [];
  pendingRemovalIds.forEach((id) => {
    if (_unmount.has(id)) return;
    const t = window.setTimeout(() => {
      setUnmount((prev) => {
        if (prev.has(id)) return prev;
        const next = new Set(prev);
        next.add(id);
        return next;
      });
    }, 180);
    timers.push(t);
  });
  return () => {
    timers.forEach((t) => window.clearTimeout(t));
  };
}, [pendingRemovalIds, _unmount]);
```
