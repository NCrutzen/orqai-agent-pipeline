3. Implement `applyLabel(page, customer_account_id, customer_name?)`:

```typescript
async function applyLabel(page: Page, customer_account_id: string, customer_name?: string) {
  await page.locator(SEL_OPEN_PICKER).click({ timeout: 8000 });
  await page.waitForSelector(SEL_PICKER_ROOT, { timeout: 8000 });
  // Search by ID first (more specific); fall back to name.
  const query = customer_account_id;
  await page.locator(SEL_SEARCH_INPUT).fill(query);
  await page.waitForTimeout(800); // type-ahead debounce per SELECTORS.md observations
  // Click the matching result row. Construct selector via the row template.
  const rowSelector = SEL_RESULT_ROW.replace("{{value}}", customer_account_id);
  await page.locator(rowSelector).first().click({ timeout: 5000 });
  if (SEL_SAVE && SEL_SAVE !== "auto-save") {
    await page.locator(SEL_SAVE).click({ timeout: 5000 });
  }
  if (SEL_CONFIRM_TOAST) {
    await page.waitForSelector(SEL_CONFIRM_TOAST, { timeout: 8000 });
  }
}
```
