- H2 `## Python` — uses LangChain's callback handler pattern:

```python
# CRITICAL: initialize the handler BEFORE creating any LangChain chains.
from orq_ai_langchain import OrqAICallbackHandler
handler = OrqAICallbackHandler(api_key=os.environ["ORQ_API_KEY"])

from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate

llm = ChatOpenAI(
    model="gpt-4o-2024-11-20",
    callbacks=[handler],       # attach handler on every LLM construction
)
prompt = ChatPromptTemplate.from_messages([("user", "{q}")])
chain = prompt | llm
result = chain.invoke({"q": "Hello"}, config={"callbacks": [handler]})
```

```python
# CRITICAL: configure the OTEL exporter BEFORE creating any tracers or LLM clients.
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter

exporter = OTLPSpanExporter(
    endpoint="https://api.orq.ai/v2/otel/v1/traces",
    headers={"Authorization": f"Bearer {os.environ['ORQ_API_KEY']}"},
)
provider = TracerProvider()
provider.add_span_processor(BatchSpanProcessor(exporter))
trace.set_tracer_provider(provider)

tracer = trace.get_tracer(__name__)
with tracer.start_as_current_span("agent.root") as span:
    span.set_attribute("customer_id", "acme-corp")
    # ... your LLM call here
```
